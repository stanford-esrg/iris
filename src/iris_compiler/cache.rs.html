14/// If code generation is spread across multiple crates, an intermediate
15/// representation in a file is required. This sets the outfile for a crate.
16pub(crate) fn set_crate_outfile(fp: String) {
17    if OUTFILE.lock().unwrap().is_some() {
18        panic!("Tried to set outfile twice");
19    }
20
21    // Env variable (e.g., $IRIS_HOME), $HOME, or unmodified
22    let fp = parse_filepath(&fp);
23
24    // Create or clear file
25    let path = PathBuf::from(&fp);
26    if path.exists() && std::fs::metadata(&path).unwrap().len() != 0 {
27        println!("Warning - clearing existing contents of file {}", fp);
28    }
29
30    println!("GOT OUTPUT FILE NAME: {}", fp);
31
32    let mut file = OpenOptions::new()
33        .create(true)
34        .write(true)
35        .truncate(true)
36        .open(&path)
37        .unwrap_or_else(|_| panic!("Failed to open file {}", fp));
38
39    let v = CACHED_DATA.lock().unwrap();
40    for elem in v.iter() {
41        let json = serde_json::to_string(&elem).expect("Failed to serialize input");
42        writeln!(file, "{}", json).expect("Failed to write to file");
43    }
44
45    *OUTFILE.lock().unwrap() = Some(path);
46}

48/// Push a new parsed input from a macro to memory and file (if available)
49pub(crate) fn push_input(input: ParsedInput) {
50    let outfile = OUTFILE.lock().unwrap();
51    match outfile.as_ref() {
52        Some(fp) => {
53            let mut file = OpenOptions::new()
54                .append(true)
55                .open(fp)
56                .expect("Failed to open file to append");
57            let json = serde_json::to_string(&input).expect("Failed to convert json to string");
58            writeln!(file, "{}", json).expect("Failed to append to file");
59        }
60        None => {
61            println!("Caching input in memory");
62        }
63    }
64    CACHED_DATA.lock().unwrap().push(input);
65}

67/// Read parsed data from input files (intermediate representation generated
68/// by some other crate, e.g., datatypes).
69pub(crate) fn set_input_files(fps: Vec<&str>) {
70    let mut cached = CACHED_DATA.lock().unwrap();
71    for fp in fps {
72        let fp = parse_filepath(fp);
73        let file =
74            File::open(fp.clone()).unwrap_or_else(|_| panic!("Cannot find input file: {}", fp));
75        let reader = BufReader::new(file);
76        for line in reader.lines() {
77            let line =
78                line.unwrap_or_else(|_| panic!("Failed to read line from input file {}", fp));
79            if line.trim().is_empty() {
80                continue;
81            }
82            let inp: ParsedInput = serde_json::from_str(&line).unwrap();
83            cached.push(inp);
84        }
85        println!("Got input from {}", fp);
86    }
87}

89fn parse_filepath(fp: &str) -> String {
90    if let Some(env_var) = fp.strip_prefix("$") {
91        let (home, path) = env_var
92            .split_once('/')
93            .map(|(v, r)| (v, Some(r)))
94            .unwrap_or((env_var, None));
95        let value =
96            std::env::var(home).unwrap_or_else(|_| panic!("Cannot find env variable ${}", home));
97
98        match path {
99            Some(path) => format!("{}/{}", value, path),
100            None => value,
101        }
102    } else if fp.starts_with("~") {
103        let home = std::env::var("HOME").expect("Cannot find home directory");
104        fp.replace("~", &home)
105    } else {
106        fp.to_string()
107    }
