22pub(crate) fn cb_to_tokens(
23    sub: &SubscriptionDecoder,
24    datatypes: &[String],
25    cb_name: &str,
26    cb_group: Option<&str>,
27    invoke_once: bool,
28    node: Option<&PNode>,
29) -> proc_macro2::TokenStream {
30    let mut conditions = vec![];
31    let mut cond_match = vec![];
32    let mut params = vec![];
33    params_to_tokens(
34        sub,
35        datatypes,
36        &mut conditions,
37        &mut cond_match,
38        &mut params,
39        node,
40    );
41    let cb_wrapper_str = cb_name.to_lowercase();
42    let cb_name = match cb_name.contains("::") {
43        // Workaround for weird formatting of CallbackSpec in PNodes; could be cleaner
44        true => cb_name.split("::").last().unwrap(),
45        false => cb_name,
46    };
47    let cb_name_ident = Ident::new(cb_name, Span::call_site());
48
49    let invoke = match cb_group {
50        Some(grp) => {
51            // Allow for unsubscribe
52            // Need to check for `is_active` in the `update` method
53            // Invoke method on struct and check for unsubscribe
54            let cb_wrapper_ident = Ident::new(&grp.to_lowercase(), Span::call_site());
55            let invoke = if grp != cb_name {
56                // Stateful callback; invoke on struct
57                quote! { conn.tracked.#cb_wrapper_ident.callback.#cb_name_ident(#( #params ), *) }
58            } else {
59                // Stateless streaming
60                quote! { #cb_name_ident(#( #params ), *) }
61            };
62            quote! {
63                if conn.tracked.#cb_wrapper_ident.is_active() {
64                    if !#invoke {
65                        conn.tracked.#cb_wrapper_ident.set_inactive();
66                        ret = true; // CB unsubscribed; something changed
67                    }
68                }
69            }
70        }
71        None => {
72            quote! { #cb_name_ident(#( #params ), *); }
73        }
74    };
75
76    let invoke = match cond_match.is_empty() {
77        true => {
78            quote! { #invoke }
79        }
80        false => {
81            quote! {
82                match ( #( #conditions ), * ) {
83                    (#( #cond_match ), *) => {
84                        #invoke
85                    }
86                    _ => {},
87                }
88            }
89        }
90    };
91
92    // Invoking this CB requires checking that it
93    // hasn't already been invoked
94    match invoke_once {
95        true => {
96            let cb_wrapper_ident = Ident::new(&cb_wrapper_str, Span::call_site());
97            quote! {
98                if conn.tracked.#cb_wrapper_ident.should_invoke() {
99                    conn.tracked.#cb_wrapper_ident.set_invoked();
100                    #invoke
101                }
102            }
103        }
104        false => quote! { #invoke },
105    }
106}

108/// Invocation of a custom filter function to tokens.
109/// This is the actual invocation of the filter predicate, either in a
110/// filter PTree (static filter functions) or in an `update` function
111/// (streaming filter functions).
112pub(crate) fn filter_func_to_tokens(
113    sub: &SubscriptionDecoder,
114    filter: &ParsedInput,
115    streaming: bool,
116) -> proc_macro2::TokenStream {
117    let needs_track = streaming || matches!(filter, ParsedInput::FilterGroupFn(_));
118    let func = match filter {
119        ParsedInput::FilterGroupFn(fil) => &fil.func,
120        ParsedInput::Filter(fil) => &fil.func,
121        _ => unreachable!(),
122    };
123    let mut conditions = vec![];
124    let mut cond_match = vec![];
125    let mut params = vec![];
126    params_to_tokens(
127        sub,
128        &func.datatypes,
129        &mut conditions,
130        &mut cond_match,
131        &mut params,
132        None,
133    );
134    let func_ident = Ident::new(&func.name, Span::call_site());
135    let fil_ident = match filter {
136        ParsedInput::FilterGroupFn(fil) => {
137            Ident::new(&fil.group_name.to_lowercase(), Span::call_site())
138        }
139        ParsedInput::Filter(fil) => Ident::new(&fil.func.name, Span::call_site()),
140        _ => unreachable!(),
141    };
142
143    // Either invoke via `tracked` parent struct or just invoke directly
144    let invoke = match filter {
145        ParsedInput::FilterGroupFn(_) => {
146            quote! { conn.tracked.#fil_ident.filter.#func_ident(#( #params ), *) }
147        }
148        ParsedInput::Filter(_) => {
149            quote! { #func_ident(#( #params ), *) }
150        }
151        _ => unreachable!(),
152    };
153    // Record FilterResult - update needs to return `true` if something changes
154    // Note that this may also be invoked in a static filter that doesn't return
155    // anything; the `ret` parameter is not used in that case.
156    let mut invoke = if needs_track {
157        quote! {
158            let filter_result = #invoke;
159            ret = ret ||
160                matches!(filter_result, FilterResult::Accept | FilterResult::Drop);
161        }
162    } else {
163        quote! { #invoke }
164    };
165    // Get return value for `update` function
166    if needs_track {
167        invoke = quote! {
168            #invoke
169            conn.tracked.#fil_ident.record_result(filter_result);
170        };
171    }
172
173    // Extract parameters for filter if needed
174    if !cond_match.is_empty() {
175        invoke = quote! {
176            match ( #( #conditions ), * ) {
177                (#( #cond_match ), *) => {
178                    #invoke
179                }
180                _ => {},
181            }
182        };
183    }
184
185    if needs_track {
186        invoke = quote! {
187            if conn.tracked.#fil_ident.is_active() {
188                #invoke
189            }
190        };
191    }
192    invoke
193}

195pub(crate) fn datatype_func_to_tokens(
196    dt: &DatatypeFnSpec,
197    filtered: bool,
198) -> proc_macro2::TokenStream {
199    let param = dt
200        .func
201        .datatypes
202        .last()
203        .unwrap_or_else(|| panic!("No parameters provided in function {:?}", dt));
204    let dt_name = Ident::new(&dt.group_name.to_lowercase(), Span::call_site());
205    let fname = Ident::new(&dt.func.name, Span::call_site());
206    if BUILTIN_TYPES.iter().any(|inp| inp.name() == param) {
207        let builtin = builtin_to_tokens(param, None);
208        if filtered {
209            // Wrapped in `TrackedDataWrapper`
210            return quote! {
211                if conn.tracked.#dt_name.is_active() {
212                    conn.tracked.#dt_name.data.#fname(#builtin);
213                }
214            };
215        } else {
216            return quote! { conn.tracked.#dt_name.#fname(#builtin); };
217        }
218    }
219    panic!("Unknown param for {}: {}", dt.func.name, param);
220}

222/// Used for filters and CBs
223/// @sub Subscription decoder must already be built up
224/// @datatypes Input datatypes (name of the type, e.g., struct)
225///
226/// Some parameters require constructors and may return Option<Self>.
227/// Must provide vectors to generate the code that accounts for this.
228/// @condition: e.g.: `match (val1, val2)`
229/// @cond_match: e.g.: `Some(val1), Some(val2) => ... `
230/// @params: actual variable names
231pub(crate) fn params_to_tokens(
232    sub: &SubscriptionDecoder,
233    datatypes: &[String],
234    conditions: &mut Vec<proc_macro2::TokenStream>,
235    cond_match: &mut Vec<proc_macro2::TokenStream>,
236    params: &mut Vec<proc_macro2::TokenStream>,
237    node: Option<&PNode>,
238) {
239    for dt in datatypes {
240        let dt_metadata = sub
241            .datatypes_raw
242            .get(dt)
243            .unwrap_or_else(|| panic!("Cannot find {} in known datatypes", dt));
244        let dt_name_ident = Ident::new(&dt.to_lowercase(), Span::call_site());
245        let dt_name_ident = if sub.filtered_datatypes.contains(dt) {
246            quote! { #dt_name_ident.data }
247        } else {
248            quote! { #dt_name_ident }
249        };
250
251        // Extract directly from tracked data
252        if sub.tracked.iter().any(|tracked| &tracked.name == dt) {
253            params.push(quote! { &conn.tracked.#dt_name_ident });
254            continue;
255        }
256
257        // Built-in datatype
258        if BUILTIN_TYPES.iter().any(|inp| inp.name() == dt) {
259            let builtin = builtin_to_tokens(dt, node);
260            params.push(quote! { #builtin });
261            continue;
262        }
263
264        // Datatype constructed in-place
265        let constructor = dt_metadata.iter().find(|inp| match inp {
266            ParsedInput::DatatypeFn(spec) => {
267                matches!(spec.func.returns, FnReturn::Constructor(_))
268            }
269            _ => false,
270        });
271        if let Some(ParsedInput::DatatypeFn(spec)) = constructor {
272            constr_to_tokens(spec, conditions, cond_match, params, &dt_name_ident);
273        }
274    }
275}

277pub(crate) fn constr_to_tokens(
278    spec: &DatatypeFnSpec,
279    conditions: &mut Vec<proc_macro2::TokenStream>,
280    cond_match: &mut Vec<proc_macro2::TokenStream>,
281    params: &mut Vec<proc_macro2::TokenStream>,
282    name_ident: &proc_macro2::TokenStream,
283) {
284    let returns = match spec.func.returns {
285        FnReturn::Constructor(Constructor::Opt) => Constructor::Opt,
286        FnReturn::Constructor(Constructor::OptRef) => Constructor::OptRef,
287        FnReturn::Constructor(Constructor::Sel) => Constructor::Sel,
288        _ => unreachable!(),
289    };
290
291    // REFACTOR: decompose this to directly reference primitive/built-in data types
292    // instead of manual list of strings
293    let constructor = {
294        assert!(spec.func.datatypes.len() == 1);
295        let dt = spec
296            .func
297            .datatypes
298            .first()
299            .expect("Constructor without datatypes");
300        let param = if dt == "L4Pdu" {
301            quote! { pdu }
302        } else if dt == "Session" {
303            quote! { conn.layers[0].last_session() }
304        } else if dt == "Mbuf" {
305            quote! { &pdu.mbuf }
306        } else if dt == "SessionProto" {
307            quote! { &conn.layers[0].last_protocol() }
308        } else {
309            panic!(
310                "Invalid input for datatype constructor: {}",
311                spec.func.datatypes.first().unwrap()
312            );
313        };
314        let name = Ident::new(&spec.func.name, Span::call_site());
315        let type_name = Ident::new(&spec.group_name, Span::call_site());
316        quote! { #type_name::#name(#param) }
317    };
318
319    if matches!(returns, Constructor::Opt | Constructor::OptRef) {
320        conditions.push(quote! { #constructor, });
321        cond_match.push(quote! { Some(#name_ident), });
322        if matches!(returns, Constructor::OptRef) {
323            params.push(quote! { #name_ident });
324        } else {
325            params.push(quote! { &#name_ident });
326        }
327    } else {
328        params.push(quote! { &#constructor });
329    }
330}

332pub(crate) fn builtin_to_tokens(name: &String, node: Option<&PNode>) -> proc_macro2::TokenStream {
333    match name.as_str() {
334        "L4Pdu" => quote! { pdu },
335        "FilterStr" => {
336            let node = node.expect(
337                "FilterStr parameter not valid (streaming callback, incompatible level, data type, or filter function)"
338            );
339            let filter_str = match &node.filter_str {
340                Some(s) => s,
341                None => panic!("No filter string found for node {}", node),
342            };
343            let filter_str = syn::LitStr::new(filter_str, Span::call_site());
344            quote! { &#filter_str }
345        }
346        "StateTxData" => quote! { &tx_data },
347        "StateTransition" => quote! { &tx },
348        "Session" => quote! { conn.layers[0].last_session() },
349        "SessionProto" => quote! { &conn.layers[0].last_protocol() },
350        "CoreId" => quote! { &conn.tracked.core_id },
351        _ => panic!("Unknown builtin datatype: {}", name),
352    }
353}

355pub(crate) fn data_actions_to_tokens(actions: &DataActions) -> proc_macro2::TokenStream {
356    let mut ret = vec![quote! {}];
357    if !actions.transport.drop() {
358        let tracked = tracked_actions_to_tokens(&actions.transport);
359        ret.push(quote! { transport_actions.extend(&#tracked); });
360    }
361    if !actions.layers[0].drop() {
362        let lyrs = tracked_actions_to_tokens(&actions.layers[0]);
363        ret.push(quote! { layer0_actions.extend(&#lyrs); });
364    }
365    quote! { #( #ret )* }
366}

385pub(crate) fn update_to_tokens(
386    sub: &SubscriptionDecoder,
387    curr: &StateTransition,
388) -> proc_macro2::TokenStream {
389    let updates = match sub.updates.get(curr) {
390        Some(updates) => updates,
391        None => return quote! {},
392    };
393    let mut ret = vec![];
394    for upd in updates {
395        // Note these CBs are only here if streaming
396        match upd {
397            ParsedInput::Callback(cb) => {
398                ret.push(cb_to_tokens(
399                    sub,
400                    &cb.func.datatypes,
401                    &cb.func.name,
402                    Some(&cb.func.name),
403                    false,
404                    None,
405                ));
406            }
407            ParsedInput::CallbackGroupFn(cb) => {
408                ret.push(cb_to_tokens(
409                    sub,
410                    &cb.func.datatypes,
411                    &cb.func.name,
412                    Some(&cb.group_name),
413                    false,
414                    None,
415                ));
416            }
417            ParsedInput::Filter(_) | ParsedInput::FilterGroupFn(_) => {
418                ret.push(filter_func_to_tokens(sub, upd, curr.is_streaming()));
419            }
420            ParsedInput::DatatypeFn(dt) => {
421                let filtered = sub.filtered_datatypes.contains(&dt.group_name);
422                ret.push(datatype_func_to_tokens(dt, filtered));
423            }
424            _ => panic!("Invalid input in update list: {:?}", upd),
425        }
426    }
427    quote! {
428        #( #ret )*
429    }
430}

460pub(crate) fn tracked_clear_to_tokens(sub: &SubscriptionDecoder) -> proc_macro2::TokenStream {
461    let mut clear = vec![];
462    for tracked in &sub.tracked {
463        let impl_clear = matches!(
464            tracked.kind,
465            TrackedKind::Datatype(true) | TrackedKind::StreamCallback | TrackedKind::StreamFilter

468            continue;

473        });
474    }
475    quote! {
476        #( #clear )*
477    }

480fn tracked_to_type_tokens(tracked: &TrackedType) -> proc_macro2::TokenStream {
481    let type_raw = Ident::new(&tracked.name, Span::call_site());
482    match tracked.kind {
483        TrackedKind::StreamCallback => {
484            quote! {
485                iris_core::subscription::callback::StreamCallbackWrapper::<#type_raw>
486            }
487        }
488        TrackedKind::StatelessCallback => {
489            quote! {
490                iris_core::subscription::callback::StatelessCallbackWrapper
491            }
492        }
493        TrackedKind::StaticCallback => {
494            quote! {
495                iris_core::subscription::callback::StaticCallbackWrapper
496            }
497        }
498        TrackedKind::StreamFilter => {
499            quote! {
500                iris_core::subscription::filter::StreamFilterWrapper::<#type_raw>
501            }
502        }
503        TrackedKind::StatelessFilter => {
504            quote! {
505                iris_core::subscription::filter::StatelessFilterWrapper
506            }
507        }
508        TrackedKind::Datatype(true) => {
509            quote! {
510                iris_core::subscription::data::TrackedDataWrapper::<#type_raw>
511            }
512        }
513        TrackedKind::Datatype(false) | TrackedKind::StaticData => quote! { #type_raw }, // No wrapper
514    }
515}

517pub(crate) fn parsers_to_tokens(sub: &SubscriptionDecoder) -> proc_macro2::TokenStream {
518    let mut parsers = vec![];
519    if sub.parsers.is_empty() {
520        println!(
521            "No session-level parsers registered; L5-L7 protocols will not be detected or parsed."
522        );
523    } else {
524        println!(
525            "Parsers: {}\n",
526            sub.parsers.iter().cloned().collect::<Vec<_>>().join(", ")
527        );
528    }
529    for parser in &sub.parsers {
530        let name = LitStr::new(parser.trim(), Span::call_site());
531        parsers.push(quote! {
532            #name
533        });
534    }
535    quote! {
536        Vec::from([#( #parsers ),*])
537    }
538}

540/// Generates tokens for streaming updates to all tracked datatypes, callbacks,
541/// and filters that need to be invoked within each streaming state.
542/// This invokes the `update` for each.
543pub(crate) fn tracked_update_to_tokens(sub: &SubscriptionDecoder) -> proc_macro2::TokenStream {
544    let mut all_updates = vec![];
545    for (level, inps) in &sub.updates {
546        if inps.is_empty() || !level.is_streaming() {
547            continue;
548        }
549        let updates = update_to_tokens(sub, level);
550        let level_ident = Ident::new(&level.to_string(), Span::call_site());
551        all_updates.push(quote! {
552            iris_core::StateTransition::#level_ident => {
553                #updates
554            }
555        });
556    }
557    quote! {
558        match tx {
559            #( #all_updates )*
560            _ => {},
561        }
562    }
563}

565/// Callback invocation to tokens within a filter PTree.
566/// This actually invokes the callback.
567pub(crate) fn fil_callback_to_tokens(
568    sub: &SubscriptionDecoder,
569    spec: &CallbackSpec,
570    node: Option<&PNode>,
571) -> proc_macro2::TokenStream {
572    let dts: Vec<String> = spec.datatypes.iter().map(|dt| dt.name.clone()).collect();
573    let name = spec.as_str.as_str();
574
575    // Will track CB state within a streaming wrapper, either because
576    // this is a multi-function CB or because it's a streaming CB
577    let mut group: Option<&str> = None;
578    if spec.subscription_id != spec.as_str {
579        group = Some(&spec.subscription_id);
580    }
581
582    // For now, streaming CBs are invoked in "update" methods only.
583    // NICE-TO-HAVE: cache past data and stream it on first match, if applicable.
584    // (In general, callback won't be invoked until filter matches)
585    let mut invoke = quote! {};
586    if match spec.expl_level {
587        Some(l) => !l.is_streaming(),
588        None => true,
589    } {
590        invoke = cb_to_tokens(sub, &dts, name, group, spec.invoke_once, node);
591    }
592    if spec.is_streaming() || spec.is_grouped() {
593        let set_active = cb_set_active_to_tokens(group.unwrap_or(name));
594        invoke = quote! {
595            #set_active
596            #invoke
597        };
598    }
599
600    // Note: CB is set as "active" (i.e., matched) based on presence in "matched" data of PTree
601
602    invoke
603}

619/// Custom streaming (stateful or stateless) predicate to tokens
620/// in a filter PTree. This checks for a filter result, but it does not
621/// invoke the filter.
622pub(crate) fn custom_pred_to_tokens(
623    name: &String,
624    matched: bool,
625    decoder: &SubscriptionDecoder,
626) -> proc_macro2::TokenStream {
627    let filter = decoder
628        .filters_raw
629        .get(name)
630        .unwrap_or_else(|| panic!("Can't find filter {}", name));
631
632    assert!(!filter.is_empty(), "{} has no inputs", name);
633
634    if filter.len() > 1 || filter[0].levels().iter().any(|l| l.is_streaming()) {
635        let ident = Ident::new(&(name.to_lowercase()), Span::call_site());
636        match matched {
637            true => quote! { conn.tracked.#ident.matched() },
638            false => quote! { conn.tracked.#ident.is_active() },
639        }
640    } else {
641        let tokens = filter_func_to_tokens(decoder, &filter[0], false);
642        quote! { #tokens == FilterResult::Accept }
643    }
644}

654pub(crate) fn layerstate_to_tokens(
655    layer: &SupportedLayer,
656    state: &LayerState,
657    op: BinOp,
658) -> proc_macro2::TokenStream {
659    let op = match op {
660        BinOp::Eq => quote! { == },
661        BinOp::Ge => quote! { >= },
662        BinOp::Gt => quote! { > },
663        BinOp::Le => quote! { <= },
664        BinOp::Lt => quote! { < },
665        _ => panic!("Invalid op for state: {:?}", op),
666    };
667    let layer_access = match layer {
668        SupportedLayer::L4 => {
669            quote! { conn.linfo.state }
670        }
671        SupportedLayer::L7 => {
672            quote! { conn.layers[0].layer_info().state }
673        }
674    };
675    let state_ident = format!("{:?}", state);
676    let state_ident = Ident::new(&state_ident, Span::call_site());
677    quote! {
678        #layer_access #op iris_core::conntrack::LayerState::#state_ident
679    }
680}

682// Binary filter predicates
683pub(crate) fn binary_to_tokens(
684    protocol: &ProtocolName,
685    field: &FieldName,
686    op: &BinOp,
687    value: &Value,
688    statics: &mut HashMap<String, (String, proc_macro2::TokenStream)>,
689) -> proc_macro2::TokenStream {
690    assert!(!field.is_combined()); // should have been split when building tree
691    let proto = Ident::new(protocol.name(), Span::call_site());
692    let field = Ident::new(field.name(), Span::call_site());
693
694    match value {
695        Value::Int(val) => {
696            let val_lit = syn::LitInt::new(&val.to_string(), Span::call_site());
697            match *op {
698                BinOp::Eq => quote! { #proto.#field() == #val_lit },
699                BinOp::Ne => quote! { #proto.#field() != #val_lit },
700                BinOp::Ge => quote! { #proto.#field() >= #val_lit },
701                BinOp::Le => quote! { #proto.#field() <= #val_lit },
702                BinOp::Gt => quote! { #proto.#field() > #val_lit },
703                BinOp::Lt => quote! { #proto.#field() < #val_lit },
704                _ => panic!("Invalid binary operation `{}` for value: `{}`.", op, value),
705            }
706        }
707        Value::IntRange { from, to } => {
708            let from_lit = syn::LitInt::new(&from.to_string(), Span::call_site());
709            let to_lit = syn::LitInt::new(&to.to_string(), Span::call_site());
710            match *op {
711                BinOp::In => quote! {
712                    #proto.#field() >= #from_lit && #proto.#field() <= #to_lit
713                },
714                _ => panic!("Invalid binary operation `{}` for value: `{}`.", op, value),
715            }
716        }
717        Value::Ipv4(ipv4net) => {
718            let addr_u32 = u32::from(ipv4net.addr());
719            let addr_lit = syn::LitInt::new(&addr_u32.to_string(), Span::call_site());
720
721            let netmask_u32 = u32::from(ipv4net.netmask());
722            let netmask_lit = syn::LitInt::new(&netmask_u32.to_string(), Span::call_site());
723
724            let net_u32 = addr_u32 & netmask_u32;
725            let net_lit = syn::LitInt::new(&net_u32.to_string(), Span::call_site());
726
727            match *op {
728                BinOp::Eq => {
729                    if ipv4net.prefix_len() == 32 {
730                        quote! { u32::from(#proto.#field()) == #addr_lit }
731                    } else {
732                        quote! { u32::from(#proto.#field()) & #netmask_lit == #net_lit }
733                    }
734                }
735                BinOp::Ne => {
736                    if ipv4net.prefix_len() == 32 {
737                        quote! { u32::from(#proto.#field()) != #addr_lit }
738                    } else {
739                        quote! { u32::from(#proto.#field()) & #netmask_lit != #net_lit }
740                    }
741                }
742                BinOp::In => {
743                    if ipv4net.prefix_len() == 32 {
744                        quote! { u32::from(#proto.#field()) == #addr_lit }
745                    } else {
746                        quote! { u32::from(#proto.#field()) & #netmask_lit == #net_lit }
747                    }
748                }
749                _ => panic!("Invalid binary operation `{}` for value: `{}`.", op, value),
750            }
751        }
752        Value::Ipv6(ipv6net) => {
753            let addr_u128 = u128::from(ipv6net.addr());
754            let addr_lit = syn::LitInt::new(&addr_u128.to_string(), Span::call_site());
755
756            let netmask_u128 = u128::from(ipv6net.netmask());
757            let netmask_lit = syn::LitInt::new(&netmask_u128.to_string(), Span::call_site());
758
759            let net_u128 = addr_u128 & netmask_u128;
760            let net_lit = syn::LitInt::new(&net_u128.to_string(), Span::call_site());
761
762            match *op {
763                BinOp::Eq => {
764                    if ipv6net.prefix_len() == 128 {
765                        quote! { u128::from(#proto.#field()) == #addr_lit }
766                    } else {
767                        quote! { u128::from(#proto.#field()) & #netmask_lit == #net_lit }
768                    }
769                }
770                BinOp::Ne => {
771                    if ipv6net.prefix_len() == 128 {
772                        quote! { u128::from(#proto.#field()) != #addr_lit }
773                    } else {
774                        quote! { u128::from(#proto.#field()) & #netmask_lit != #net_lit }
775                    }
776                }
777                BinOp::In => {
778                    if ipv6net.prefix_len() == 128 {
779                        quote! { u128::from(#proto.#field()) == #addr_lit }
780                    } else {
781                        quote! { u128::from(#proto.#field()) & #netmask_lit == #net_lit }
782                    }
783                }
784                _ => panic!("Invalid binary operation `{}` for value: `{}`.", op, value),
785            }
786        }
787        Value::Text(text) => match *op {
788            BinOp::Eq => {
789                let val_lit = syn::LitStr::new(text, Span::call_site());
790                quote! { #proto.#field() == #val_lit }
791            }
792            BinOp::Ne => {
793                let val_lit = syn::LitStr::new(text, Span::call_site());
794                quote! { #proto.#field() != #val_lit }
795            }
796            BinOp::En => {
797                let field_ident = Ident::new(&field.to_string().to_camel_case(), Span::call_site());
798                let variant_ident = Ident::new(&text.as_str().to_camel_case(), Span::call_site());
799                quote! { #proto.#field() == iris_core::protocols::stream::#proto::#field_ident::#variant_ident }
800            }
801            BinOp::Re => {
802                if Regex::new(text).is_err() {
803                    panic!("Invalid Regex string: {}", text);
804                }
805                let val_lit = syn::LitStr::new(text, Span::call_site());
806                let kind = quote! { regex::Regex };
807                let re_ident = static_ident_re(statics, text, val_lit, kind);
808                quote! {
809                    #re_ident.is_match(&#proto.#field()[..])
810                }
811            }
812            BinOp::ByteRe => {
813                if BytesRegex::new(text).is_err() {
814                    panic!("Invalid Regex string: {}", text);
815                }
816                let val_lit = syn::LitStr::new(text, Span::call_site());
817                let kind = quote! { regex::bytes::Regex };
818                let re_ident = static_ident_re(statics, text, val_lit, kind);
819                quote! {
820                    #re_ident.is_match((&#proto.#field()).as_ref())
821                }
822            }
823            BinOp::Contains => {
824                let val_lit = syn::LitStr::new(text, Span::call_site());
825                let finder_ident =
826                    static_ident_memchr(statics, text, quote! { #val_lit.as_bytes() });
827                quote! {
828                    #finder_ident.find(#proto.#field().as_bytes()).is_some()
829                }
830            }
831            BinOp::NotContains => {
832                let val_lit = syn::LitStr::new(text, Span::call_site());
833                let finder_ident =
834                    static_ident_memchr(statics, text, quote! { #val_lit.as_bytes() });
835                quote! {
836                    #finder_ident.find(#proto.#field().as_bytes()).is_none()
837                }
838            }
839            _ => panic!("Invalid binary operation `{}` for value: `{}`.", op, value),
840        },
841        Value::Byte(b) => match *op {
842            BinOp::Eq => {
843                let bytes_lit = syn::LitByteStr::new(b, Span::call_site());
844                quote! {
845                    #proto.#field().as_ref() as &[u8] == #bytes_lit
846                }
847            }
848            BinOp::Ne => {
849                let bytes_lit = syn::LitByteStr::new(b, Span::call_site());
850                quote! {
851                    #proto.#field().as_ref() as &[u8] != #bytes_lit
852                }
853            }
854            BinOp::Contains => {
855                let bytes_lit = syn::LitByteStr::new(b, Span::call_site());
856                let debug = b.iter().map(|b| format!("{:02x}", b)).collect::<String>();
857                let finder_ident = static_ident_memchr(statics, &debug, quote! { #bytes_lit });
858                quote! {
859                    #finder_ident.find(#proto.#field().as_ref()).is_some()
860                }
861            }
862            BinOp::NotContains => {
863                let bytes_lit = syn::LitByteStr::new(b, Span::call_site());
864                let debug = b.iter().map(|b| format!("{:02x}", b)).collect::<String>();
865                let finder_ident = static_ident_memchr(statics, &debug, quote! { #bytes_lit });
866                quote! {
867                    #finder_ident.find(#proto.#field().as_ref()).is_none()
868                }
869            }
870            _ => panic!("Invalid binary operation `{}` for value: `{}`.", op, value),
871        },
872    }
873}
874
875fn static_ident_re(
876    statics: &mut HashMap<String, (String, proc_macro2::TokenStream)>,
877    text: &String,
878    val_lit: syn::LitStr,
879    kind: proc_macro2::TokenStream,
880) -> Ident {
881    let key = format!("RE_{}", text);
882    match statics.get(&key) {
883        Some((name, _)) => Ident::new(name, Span::call_site()),
884        None => {
885            let name = format!("RE{}", statics.len());
886            let ident = Ident::new(&name, Span::call_site());
887            let lazy = quote! {
888                static ref #ident: #kind = #kind::new(#val_lit).unwrap();
889            };
890            statics.insert(key, (name, lazy));
891            ident
892        }
893    }
894}
895
896fn static_ident_memchr(
897    statics: &mut HashMap<String, (String, proc_macro2::TokenStream)>,
898    text: &String,
899    val_lit: proc_macro2::TokenStream,
900) -> Ident {
901    let key = format!("FINDER_{}", text);
902    match statics.get(&key) {
903        Some((name, _)) => Ident::new(name, Span::call_site()),
904        None => {
905            let name = format!("FINDER{}", statics.len());
906            let ident = Ident::new(&name, Span::call_site());
907            let lazy = quote! {
908                static ref #ident: memchr::memmem::Finder<'static> = memchr::memmem::Finder::new(#val_lit);
909            };
910            statics.insert(key, (name, lazy));
911            ident
912        }
913    }
