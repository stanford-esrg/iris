64pub(crate) struct SubscriptionSpec {
65    pub(crate) callbacks: Vec<CallbackSpec>,
66    pub(crate) filter: String,
67    pub(crate) as_str: String,
68    pub(crate) patterns: Option<Vec<FlatPattern>>,
69}

71impl SubscriptionSpec {
72    fn add_patterns(&mut self, custom_preds: &[Predicate]) {
73        if self.patterns.is_none() {
74            let filter = Filter::new(&self.filter, custom_preds)
75                .unwrap_or_else(|_| panic!("Invalid filter: {}", self.filter));
76            self.patterns = Some(filter.get_patterns_flat());
77        }
78        let filter_str = self
79            .callbacks
80            .iter()
81            .any(|cb| cb.datatypes.iter().any(|dt| dt.name == "FilterStr"));
82        if filter_str && let Some(pats) = &mut self.patterns {
83            for pat in pats {
84                pat.as_str = Some(pat.to_string());
85            }
86        }
87    }
88
89    fn get_filter_parsers(&self) -> HashSet<String> {
90        let mut ret = HashSet::new();
91        for pat in self.patterns.as_ref().unwrap() {
92            for pred in &pat.predicates {
93                if pred.on_proto() {
94                    ret.insert(pred.get_protocol().clone().0);
95                }
96            }
97        }
98        ret
99    }
100
101    fn add_invoke_once(&mut self) {
102        // We don't make guarantees for grouped CBs, since we track grouped CBs as a full group.
103        // NICE-TO-HAVE: could change this in the future.
104        if self.callbacks.len() > 1 {
105            return;
106        }
107
108        // Streaming CBs invoked until unsubscribe; L4Terminated subscriptions
109        // don't need to be tracked.
110        if let Some(level) = self.callbacks[0].expl_level
111            && (level.is_streaming() || level == StateTransition::L4Terminated)
112        {
113            return;
114        }
115
116        // Any CB that has streaming patterns
117        let patterns = self.patterns.as_ref().expect("Patterns not populated");
118        let streaming_pred = patterns
119            .iter()
120            .any(|pat| pat.predicates.iter().any(|pred| pred.is_streaming()));
121
122        if streaming_pred {
123            for cb in &mut self.callbacks {
124                cb.invoke_once = true;
125            }
126        }
127    }
128}

130/// Responsible for transforming the raw data in `parse.rs` into the formats
131/// Iris requires.
132pub(crate) struct SubscriptionDecoder {
133    /// Filter group (or function name) --> Parsed Input(s)
134    pub(crate) filters_raw: HashMap<String, Vec<ParsedInput>>,
135    /// Map datatype group (or name) -> Parsed Input(s)
136    pub(crate) datatypes_raw: HashMap<String, Vec<ParsedInput>>,
137    /// Map cb group (or name) -> Parsed Input(s)
138    pub(crate) cbs_raw: HashMap<String, Vec<ParsedInput>>,
139
140    /// Required stream protocol parsers
141    pub(crate) parsers: HashSet<String>,
142
143    /// Valid custom predicates passed into Filter::new()
144    pub(crate) custom_preds: Vec<Predicate>,
145    /// Datatype name -> Datatype Spec with all updates
146    /// Used to derive levels for callbacks and custom filters
147    pub(crate) datatypes: HashMap<String, StateTransitionSpec>,
148    /// Datatypes marked explicitly as `tracked`; these are wrapped at runtime
149    /// to discard if the associated subscriptions go out of scope, under the assumption
150    /// that they are computationally and/or memory intensive.
151    pub(crate) filtered_datatypes: HashSet<String>,
152    /// Full subscriptions
153    pub(crate) subscriptions: Vec<SubscriptionSpec>,
154
155    /// Required `updates`: Level of required update -->
156    /// datatype update, filter method, or streaming callback.
157    pub(crate) updates: HashMap<StateTransition, Vec<ParsedInput>>,
158    /// Tracked datatypes (stored as fields in Tracked struct)
159    pub(crate) tracked: HashSet<TrackedType>,
160}

162impl SubscriptionDecoder {
163    pub(crate) fn new(inputs: &Vec<ParsedInput>) -> Self {
164        let mut ret = Self {
165            filters_raw: HashMap::new(),
166            datatypes_raw: HashMap::new(),
167            cbs_raw: HashMap::new(),
168            parsers: HashSet::new(),
169            custom_preds: Vec::new(),
170            datatypes: HashMap::new(),
171            filtered_datatypes: HashSet::new(),
172            subscriptions: Vec::new(),
173            updates: HashMap::new(),
174            tracked: HashSet::new(),
175        };
176        ret.parse_raw(inputs);
177        ret.decode_datatypes();
178        ret.decode_filters();
179        ret.decode_subscriptions();
180        ret.prune_filters_and_datatypes();
181        ret.add_parsers();
182        for spec in &mut ret.subscriptions {
183            spec.add_patterns(&ret.custom_preds);
184            ret.parsers.extend(spec.get_filter_parsers());
185            spec.add_invoke_once();
186        }
187        ret.decode_updates();
188        assert!(!ret.cbs_raw.is_empty(), "No callbacks defined");
189        ret
190    }

192    /// Map inputs by name
193    fn parse_raw(&mut self, inputs: &Vec<ParsedInput>) {
194        BUILTIN_TYPES.iter().for_each(|dt| {
195            self.datatypes_raw
196                .insert(dt.name().clone(), vec![dt.clone()]);
197        });
198        for inp in inputs {
199            let name = inp.name().clone();
200            match inp {
201                ParsedInput::Datatype(_) => {
202                    let v = self.datatypes_raw.entry(name).or_insert(vec![]);
203                    v.push(inp.clone());
204                }
205                ParsedInput::DatatypeFn(dt) => {
206                    let group_name = dt.group_name.clone();
207                    let v = self.datatypes_raw.entry(group_name).or_insert(vec![]);
208                    v.push(inp.clone());
209                }
210                ParsedInput::Filter(_) => {
211                    self.filters_raw.insert(name.clone(), vec![inp.clone()]);
212                }
213                ParsedInput::FilterGroup(_) => {
214                    let v = self.filters_raw.entry(name).or_insert(vec![]);
215                    v.push(inp.clone());
216                }
217                ParsedInput::FilterGroupFn(f) => {
218                    let group_name = f.group_name.clone();
219                    let v = self.filters_raw.entry(group_name).or_insert(vec![]);
220                    v.push(inp.clone());
221                }
222                ParsedInput::Callback(_) => {
223                    assert!(
224                        !self.cbs_raw.contains_key(&name),
225                        "Callback {} defined twice",
226                        name
227                    );
228                    self.cbs_raw.insert(name.clone(), vec![inp.clone()]);
229                }
230                ParsedInput::CallbackGroup(_) => {
231                    let v = self.cbs_raw.entry(name).or_insert(vec![]);
232                    v.push(inp.clone());
233                }
234                ParsedInput::CallbackGroupFn(cb) => {
235                    let group_name = cb.group_name.clone();
236                    let v = self.cbs_raw.entry(group_name).or_insert(vec![]);
237                    v.push(inp.clone());
238                }
239            }
240        }
241    }

243    fn decode_datatypes(&mut self) {
244        for (dt, inp) in &self.datatypes_raw {
245            let spec = StateTransitionSpec {
246                name: dt.clone(),
247                updates: inp.iter().cloned().flat_map(|l| l.levels()).collect(),
248            };
249            self.datatypes.insert(dt.clone(), spec);
250            if inp.iter().any(|i| match i {
251                ParsedInput::Datatype(d) => d.filtered,
252                _ => false,
253            }) {
254                self.filtered_datatypes.insert(dt.clone());
255            }
256        }
257    }
258
259    fn decode_filters(&mut self) {
260        for (name, v) in self.filters_raw.iter() {
261            // Validate grouped input
262            if v.iter().any(|inp| inp.is_group()) {
263                assert!(v.len() > 1, "Missing group for {:?}", v);
264                assert!(
265                    v.iter().map(|s| s.group()).all(|x| x == v[0].group()),
266                    "Mismatched groups in: {:?}",
267                    v
268                );
269            } else {
270                assert!(v.len() == 1, "Missing filter group for: {:?}", v);
271            }
272
273            // TODO this can break if filter is trying to request multiple datatypes
274            // Filter group gets "level" at max level even if there are multiple functions
275            let mut levels = vec![];
276            let mut filtered_data = vec![];
277            for inp in v {
278                let mut lvls = vec![];
279                // Levels of the datatypes in the function(s)
280                match inp {
281                    ParsedInput::Filter(f) => {
282                        lvls.extend(self.fil_datatypes_to_levels(&f.func, name));
283                    }
284                    ParsedInput::FilterGroupFn(f) => {
285                        lvls.extend(self.fil_datatypes_to_levels(
286                            &f.func,
287                            &format!("{}::{}", f.group_name, f.func.name),
288                        ));
289                    }
290                    _ => continue,
291                }
292                if lvls.iter().any(|l| l.is_streaming()) {
293                    assert!(
294                        !inp.levels().is_empty(),
295                        "Filter {}::{} requests streaming datatypes; must explicitly specify level",
296                        name,
297                        inp.name()
298                    );
299                }
300                // Explicitly annotated levels
301                lvls.extend(inp.levels());
302                lvls.sort();
303                lvls.dedup();
304                if !lvls.is_empty() {
305                    levels.push(lvls);
306                }
307                // Filtered ("expensive") datatypes that require tracking in PTree
308                match inp {
309                    ParsedInput::Filter(f) => {
310                        filtered_data.extend(
311                            f.func
312                                .datatypes
313                                .iter()
314                                .filter(|dt| self.filtered_datatypes.contains(*dt))
315                                .cloned(),
316                        );
317                    }
318                    ParsedInput::FilterGroupFn(f) => {
319                        filtered_data.extend(
320                            f.func
321                                .datatypes
322                                .iter()
323                                .filter(|dt| self.filtered_datatypes.contains(*dt))
324                                .cloned(),
325                        );
326                    }
327                    _ => continue,
328                }
329            }
330            self.custom_preds.push(Predicate::Custom {
331                name: FuncIdent(name.clone()),
332                levels,
333                matched: true,
334                filtered_data,
335            });
336        }
337    }

339    fn decode_subscriptions(&mut self) {
340        for (cb_name, v) in &self.cbs_raw {
341            let inp_group = v
342                .iter()
343                .find(|i| matches!(i, ParsedInput::Callback(_) | ParsedInput::CallbackGroup(_)))
344                .unwrap_or_else(|| panic!("{} missing callback definition", cb_name));
345            let filter = match inp_group {
346                ParsedInput::Callback(cb) => cb.filter.clone(),
347                ParsedInput::CallbackGroup(cb) => cb.filter.clone(),
348                _ => unreachable!(),
349            };
350            let mut callbacks = vec![];
351
352            for inp in v {
353                match inp {
354                    ParsedInput::Callback(cb) => {
355                        assert!(v.len() == 1);
356                        self.specs_to_cb(
357                            &cb.func,
358                            cb_name.clone(),
359                            cb_name.clone(),
360                            &cb.level,
361                            &mut callbacks,
362                        );
363                    }
364                    ParsedInput::CallbackGroupFn(cb) => {
365                        let sub_id = cb.group_name.clone();
366                        let as_str = format!("{}::{}", sub_id, cb.func.name);
367                        self.specs_to_cb(&cb.func, as_str, sub_id, &cb.level, &mut callbacks);
368                    }
369                    ParsedInput::CallbackGroup(_) => continue,
370                    _ => panic!("Unknown ParsedInput in callback list"),
371                }
372            }
373            self.subscriptions.push(SubscriptionSpec {
374                callbacks,
375                filter,
376                as_str: cb_name.clone(),
377                patterns: None,
378            });
379        }
380    }
381
382    fn specs_to_cb(
383        &self,
384        spec: &FnSpec,
385        as_str: String,
386        subscription_id: String,
387        levels: &[StateTransition],
388        callbacks: &mut Vec<CallbackSpec>,
389    ) {
390        if levels.len() <= 1 {
391            let cb_spec =
392                self.spec_to_cbs_int(spec, as_str, subscription_id, levels.last().cloned());
393            callbacks.push(cb_spec);
394        } else {
395            // Break up into multiple callbacks for multiple different levels.
396            // REFACTOR: might be simpler in the future to change CallbackSpec to take multiple levels?
397            let mut levels: Vec<StateTransition> = levels.to_vec();
398            levels.sort();
399            levels.dedup();
400            for l in levels {
401                let cb_spec =
402                    self.spec_to_cbs_int(spec, as_str.clone(), subscription_id.clone(), Some(l));
403                callbacks.push(cb_spec);
404            }
405        }
406    }

408    fn spec_to_cbs_int(
409        &self,
410        spec: &FnSpec,
411        as_str: String,
412        subscription_id: String,
413        expl_level: Option<StateTransition>,
414    ) -> CallbackSpec {
415        let must_deliver = spec.datatypes.iter().any(|dt| dt == "FilterStr");
416        let datatypes = spec
417            .datatypes
418            .iter()
419            .map(|dt_name| {
420                self.datatypes
421                    .get(dt_name)
422                    .unwrap_or_else(|| panic!("Can't find datatype {}", dt_name))
423                    .clone()
424            })
425            .collect::<Vec<_>>();
426        let filtered_data = spec
427            .datatypes
428            .iter()
429            .filter(|dt| self.filtered_datatypes.contains(*dt))
430            .cloned()
431            .collect::<Vec<_>>();
432        if datatypes
433            .iter()
434            .any(|dt| dt.updates.iter().any(|l| l.is_streaming()))
435        {
436            assert!(
437                expl_level.is_some(),
438                "Callback {} requests streaming datatype; must explicitly specify level.",
439                spec.name
440            );
441        }
442        if datatypes
443            .iter()
444            .any(|dt| dt.updates.contains(&StateTransition::L4Terminated))
445        {
446            assert!(
447                expl_level.is_none() || expl_level.unwrap() == StateTransition::L4Terminated,
448                "Callback {} requests datatype that is L4Terminated; cannot specify separate level",
449                spec.name
450            );
451        }
452        CallbackSpec {
453            expl_level,
454            datatypes,
455            must_deliver,
456            invoke_once: false, // Filled in when patterns are built
457            as_str,
458            subscription_id,
459            filtered_data,
460        }
461    }

463    // TODO figure out what should go here -- need to fix filters to have multiple levels / params?
464    // Pull the top-level datatype declaration to infer a function level
465    fn fil_datatypes_to_levels(&self, spec: &FnSpec, as_str: &String) -> Vec<StateTransition> {
466        let mut lvls = vec![];
467        for dt_name in &spec.datatypes {
468            let dt_info = self
469                .datatypes_raw
470                .get(dt_name)
471                .unwrap_or_else(|| panic!("Cannot find datatype {}", dt_name));
472            // Try level declared on struct first
473            let mut level = dt_info
474                .iter()
475                .find(|grp| matches!(grp, ParsedInput::Datatype(_)))
476                .unwrap_or_else(|| panic!("Cannot find datatype declaration {}", dt_name))
477                .levels();
478            // If no level declared on struct, check for functions
479            if level.is_empty() {
480                level = dt_info
481                    .iter()
482                    .find(|grp| matches!(grp, ParsedInput::DatatypeFn(_)))
483                    .unwrap_or_else(|| {
484                        panic!("Cannot find datatype function declaration {}", dt_name)
485                    })
486                    .levels();
487            }
488            level.sort();
489            level.dedup();
490            assert!(
491                level.len() == 1,
492                "{} declaration has {} levels (requires 1 to be used in filter function; try declaring level on struct)",
493                dt_name,
494                level.len()
495            );
496            lvls.push(level.pop().unwrap());
497        }
498        lvls.sort();
499        lvls.dedup();
500        if lvls.len() > 1 {
501            panic!(
502                "Multiple datatypes with different levels in filter function not supported: {}",
503                as_str
504            );
505        }
506        lvls
507    }

509    fn prune_filters_and_datatypes(&mut self) {
510        // Only retain filters that are actually used by a subscription
511        self.filters_raw.retain(|name, _| {
512            self.subscriptions
513                .iter()
514                .any(|spec| spec.filter.contains(name))
515        });
516
517        // Only retain datatypes that are actually used by a datatype or filter
518        let mut req_datatypes = HashSet::new();
519        for v in self.filters_raw.values() {
520            for inp in v {
521                match inp {
522                    ParsedInput::FilterGroupFn(spec) => {
523                        req_datatypes.extend(&spec.func.datatypes);
524                    }
525                    ParsedInput::Filter(spec) => {
526                        req_datatypes.extend(&spec.func.datatypes);
527                    }
528                    _ => {}
529                }
530            }
531        }
532        for v in self.cbs_raw.values() {
533            for inp in v {
534                match inp {
535                    ParsedInput::Callback(spec) => {
536                        req_datatypes.extend(&spec.func.datatypes);
537                    }
538                    ParsedInput::CallbackGroupFn(spec) => {
539                        req_datatypes.extend(&spec.func.datatypes);
540                    }
541                    _ => {}
542                }
543            }
544        }
545        self.datatypes_raw
546            .retain(|name, _| req_datatypes.contains(name));
547    }

549    fn add_parsers(&mut self) {
550        for v in self.filters_raw.values() {
551            self.parsers.extend(
552                v.iter()
553                    .flat_map(|i| i.expl_parsers())
554                    .filter(|p| !p.is_empty()),
555            );
556        }
557        for v in self.datatypes_raw.values() {
558            self.parsers.extend(
559                v.iter()
560                    .flat_map(|i| i.expl_parsers())
561                    .filter(|p| !p.is_empty()),
562            );
563        }
564        for v in self.cbs_raw.values() {
565            self.parsers.extend(
566                v.iter()
567                    .flat_map(|i| i.expl_parsers())
568                    .filter(|p| !p.is_empty()),
569            );
570        }

573    fn decode_updates(&mut self) {
574        let mut updates = HashMap::new();
575        for v in self.filters_raw.values() {
576            Self::push_update(&mut updates, v);
577            if let Some(tracked) = Self::is_tracked_type(v) {
578                self.tracked.insert(tracked);
579            }
580        }
581        for v in self.datatypes_raw.values() {
582            Self::push_update(&mut updates, v);
583            if let Some(tracked) = Self::is_tracked_type(v) {
584                self.tracked.insert(tracked);
585            }
586        }
587        for v in self.cbs_raw.values() {
588            Self::push_update(&mut updates, v);
589            if let Some(tracked) = Self::is_tracked_type(v) {
590                self.tracked.insert(tracked);
591            }
592        }
593        for spec in &self.subscriptions {
594            for cb in &spec.callbacks {
595                if cb.invoke_once {
596                    assert!(
597                        cb.subscription_id == cb.as_str,
598                        "Grouped CB {}::{} marked as invoke_once",
599                        cb.subscription_id,
600                        cb.as_str
601                    );
602                    self.tracked.insert(TrackedType {
603                        name: cb.as_str.clone(),
604                        kind: TrackedKind::StaticCallback,
605                    });
606                }
607
608                // Datatypes that are not builtin but need
609                // to be cached for future delivery
610                // TODO - change this to be datatypes that are ready before
611                // callbacks and shouldn't be constructed in-place.
612                // TODO - change to only do this if the subscription can't
613                // be delivered at L4FirstPacket
614                let static_dts = cb
615                    .datatypes
616                    .iter()
617                    .filter(|dt| {
618                        dt.updates.len() == 1 && dt.updates[0] == StateTransition::L4FirstPacket
619                    })
620                    .collect::<Vec<_>>();
621                for dt in static_dts {
622                    self.tracked.insert(TrackedType {
623                        name: dt.name.clone(),
624                        kind: TrackedKind::StaticData,
625                    });
626                }
627            }
628        }
629        self.updates = updates;
630    }

632    fn push_update(
633        updates: &mut HashMap<StateTransition, Vec<ParsedInput>>,
634        inps: &Vec<ParsedInput>,
635    ) {
636        for inp in inps {
637            if inp.is_group() {
638                continue;
639            }
640            let (streaming, stat): (Vec<_>, Vec<_>) =
641                inp.levels().into_iter().partition(|l| l.is_streaming());
642            for l in streaming {
643                updates.entry(l).or_insert(vec![]).push(inp.clone());
644            }
645            for l in stat {
646                match inp {
647                    // Deliver requested tx update to filter and datatype fns
648                    ParsedInput::FilterGroupFn(_) => {
649                        updates.entry(l).or_insert(vec![]).push(inp.clone());
650                    }
651                    ParsedInput::DatatypeFn(f)
652                        if !matches!(f.func.returns, FnReturn::Constructor(_)) =>
653                    {
654                        updates.entry(l).or_insert(vec![]).push(inp.clone());
655                    }
656                    // Callbacks invoked inline
657                    // Ungrouped filter functions that aren't streaming don't need to be tracked
658                    _ => {}
659                }
660            }
661        }
662    }

664    fn is_tracked_type(inps: &Vec<ParsedInput>) -> Option<TrackedType> {
665        let is_streaming = inps
666            .iter()
667            .any(|inp| inp.levels().iter().any(|l| l.is_streaming()));
668        // Only one function and it's not streaming
669        if inps.len() == 1 && inps[0].levels().len() == 1 && !is_streaming {
670            return None;
671        }
672        // Multiple parsed inputs, but one is a group definition and
673        // the other is a constructor.
674        if !is_streaming {
675            let fns = inps
676                .iter()
677                .filter(|i| {
678                    matches!(
679                        i,
680                        ParsedInput::CallbackGroupFn(_)
681                            | ParsedInput::DatatypeFn(_)
682                            | ParsedInput::FilterGroupFn(_)
683                    )
684                })
685                .collect::<Vec<_>>();
686            if fns.is_empty() {
687                return None;
688            }
689            if fns.len() == 1 && fns.last().unwrap().is_constructor() {
690                return None;
691            }
692        }
693
694        // Streaming or multiple functions
695        for inp in inps {
696            match inp {
697                ParsedInput::CallbackGroup(group) => {
698                    return Some(TrackedType {
699                        kind: TrackedKind::StreamCallback,
700                        name: group.name.clone(),
701                    });
702                }
703                ParsedInput::FilterGroup(group) => {
704                    return Some(TrackedType {
705                        kind: TrackedKind::StreamFilter,
706                        name: group.name.clone(),
707                    });
708                }
709                ParsedInput::Datatype(group) => {
710                    return Some(TrackedType {
711                        kind: TrackedKind::Datatype(group.filtered),
712                        name: group.name.clone(),
713                    });
714                }
715                ParsedInput::Callback(cb) => {
716                    return Some(TrackedType {
717                        kind: TrackedKind::StatelessCallback,
718                        name: cb.func.name.clone(),
719                    });
720                }
721                ParsedInput::Filter(fil) => {
722                    return Some(TrackedType {
723                        kind: TrackedKind::StatelessFilter,
724                        name: fil.func.name.clone(),
725                    });
726                }
727                _ => continue,
728            }
729        }
730        None
731    }

760    /// The application requires a filter at this stage
761    /// I.e., at this state transition, the application may have
762    /// new information to invoke a callback, update Actions, or
763    /// drop traffic.
764    pub(crate) fn requires_filter(&self, filter_layer: &StateTransition) -> bool {
765        for spec in &self.subscriptions {
766            for cb in &spec.callbacks {
767                let patterns = spec.patterns.as_ref().expect("Patterns not populated");
768                for pat in patterns {
769                    let level = SubscriptionLevel::new(&cb.datatypes, pat, cb.expl_level);
770                    if level.can_skip(filter_layer) {
771                        continue;
772                    }
773                    // 1. Delivery may be required or no longer required
774                    // (in case of unsubscribe)
775                    if level.can_deliver(filter_layer) {
776                        return true;
777                    }
778                    // 2. Callback might unsubscribe
779                    if let Some(cb) = level.callback
780                        && cb.is_streaming()
781                        && &cb == filter_layer
782                    {
783                        return true;
784                    }
785                    // 3. Filter predicate might match
786                    if level.filter_preds.iter().any(|l| l == filter_layer) {
787                        return true;
788                    }
789                }
790                for dt in &cb.datatypes {
791                    if dt.updates.iter().any(|l| l == filter_layer) {
792                        return true;
793                    }
794                }
795            }
796        }
797        false
798    }

801#[derive(Debug, Clone, PartialEq, Eq)]
802pub(crate) enum TrackedKind {
803    StreamCallback,
804    StatelessCallback,
805    StaticCallback,
806    StreamFilter,
807    StatelessFilter,
808    // True if the datatype is filtered (requires tracking in PTree)
809    Datatype(bool),
810    // Stored in struct but does not require updates
811    StaticData,
812}
813
814#[derive(Debug, Clone, PartialEq, Eq)]
815pub(crate) struct TrackedType {
816    pub(crate) kind: TrackedKind,
817    pub(crate) name: String,
818}

823        self.name.hash(state);
824        format!("{:?}", self.kind).hash(state);
825    }
