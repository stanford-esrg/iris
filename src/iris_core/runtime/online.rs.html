28impl<S> OnlineRuntime<S>
29where
30    S: Subscribable,
31{
32    pub(crate) fn new(
33        config: &RuntimeConfig,
34        options: OnlineOptions,
35        mempools: &mut BTreeMap<SocketId, Mempool>,
36        hw_filter_str: String,
37        subscription: Arc<Subscription<S>>,
38    ) -> Self {
39        let hw_filter = Filter::new(&hw_filter_str, &[]).expect("Failed to parse collapsed filter");
40        // Set up signal handler
41        let is_running = Arc::new(AtomicBool::new(true));
42        let r = Arc::clone(&is_running);
43        ctrlc::set_handler(move || {
44            r.store(false, Ordering::Relaxed);
45        })
46        .expect("Error setting Ctrl-C handler");
47
48        log::info!("Initializing Ports...");
49        let mut ports: BTreeMap<PortId, Port> = BTreeMap::new();
50        for port_map in options.online.ports.iter() {
51            let port = Port::new(port_map);
52            let socket_id = port.id.socket_id();
53            mempools.entry(socket_id).or_insert_with(|| {
54                // Create a local mempool if user is not polling the port
55                // from the same socket.
56                let mtu = if let Some(online) = &config.online {
57                    online.mtu
58                } else {
59                    Mempool::default_mtu()
60                };
61                Mempool::new(&config.mempool, socket_id, mtu)
62                    .expect("Unable to initialize local mempool")
63            });
64            port.init(
65                mempools,
66                options.online.nb_rxd,
67                options.online.mtu,
68                options.online.promiscuous,
69            )
70            .expect("Failed to initialize port.");
71            ports.insert(port.id, port);
72        }
73
74        log::info!("Initializing RX Cores...");
75        let mut rx_cores: BTreeMap<CoreId, RxCore<S>> = BTreeMap::new();
76        let mut core_map: BTreeMap<CoreId, Vec<RxQueue>> = BTreeMap::new();
77        for (_port_id, port) in ports.iter() {
78            for (rxqueue, core_id) in port.queue_map.iter() {
79                core_map.entry(*core_id).or_default().push(*rxqueue);
80            }
81        }
82        for (core_id, rxqueues) in core_map.into_iter() {
83            let rx_core = RxCore::new(
84                core_id,
85                rxqueues,
86                options.conntrack.clone(),
87                #[cfg(feature = "prometheus")]
88                options.online.prometheus.is_some(),
89                Arc::clone(&subscription),
90                Arc::clone(&is_running),
91            );
92            rx_cores.insert(core_id, rx_core);
93        }
94
95        let monitor = Some(Monitor::new(config, &ports, Arc::clone(&is_running)));
96
97        OnlineRuntime {
98            ports,
99            rx_cores,
100            monitor,
101            filter: hw_filter,
102            options,
103        }
104    }
105
106    pub(crate) fn run(&mut self) {
107        self.start_ports();
108
109        log::info!("Launching RX cores...");
110        for (core_id, _rx_core) in self.rx_cores.iter() {
111            let role = unsafe { dpdk::rte_eal_lcore_role(core_id.raw()) };
112            if role != dpdk::rte_lcore_role_t_ROLE_RTE {
113                log::error!("Attempted to launch non-DPDK core");
114                panic!();
115            }
116
117            let arg = &self.rx_cores as *const _ as *mut c_void;
118            let ret = unsafe {
119                dpdk::rte_eal_remote_launch(Some(launch_rx::<S>), arg, core_id.raw() as c_uint)
120            };
121            if ret != 0 {
122                log::error!("RX Core {} busy, launch failed.", core_id);
123                panic!();
124            }
125        }
126
127        // run main thread
128        self.run_main();
129        unsafe { dpdk::rte_eal_mp_wait_lcore() };
130
131        log::info!("Exiting loop...");
132        self.stop_ports();
133    }
134
135    fn run_main(&mut self) {
136        let id = unsafe { dpdk::rte_lcore_id() };
137        log::info!("Running main on Core {}", id);
138        let start = Instant::now();
139        let tokio_runtime = tokio::runtime::Builder::new_current_thread() // Use a current-thread runtime
140            .enable_all() // Enables all necessary features (I/O, time, etc.)
141            .build()
142            .expect("Failed to create Tokio runtime");
143        let mut monitor = self.monitor.take().unwrap();
144        tokio_runtime.block_on(async move {
145            let jh1 = tokio::spawn(async move {
146                monitor.run().await;
147            });
148            #[cfg(feature = "prometheus")]
149            if let Some(prometheus) = self.options.online.prometheus {
150                use hyper::service::service_fn;
151                use hyper_util::rt::TokioIo;
152                use tokio::net::TcpListener;
153                tokio::spawn(async move {
154                    let listener = TcpListener::bind((prometheus.ip, prometheus.port))
155                        .await
156                        .expect("failed to run prometheus server");
157                    loop {
158                        let Ok((socket, _)) = listener.accept().await else {
159                            eprintln!("Prometheus server accept failed");
160                            break;
161                        };
162                        let socket = TokioIo::new(socket);
163                        tokio::spawn(async move {
164                            if let Err(e) = hyper::server::conn::http1::Builder::new()
165                                .serve_connection(socket, service_fn(crate::stats::serve_req))
166                                .await
167                            {
168                                eprintln!("Prometheus server error: {}", e);
169                            }
170                        });
171                    }
172                });
173            }
174            jh1.await.unwrap();
175        });
176        println!("Main done. Ran for {:?}", start.elapsed());
177    }
178
179    fn start_ports(&self) {
180        log::info!("Starting ports...");
181        for port in self.ports.values() {
182            port.start();
183
184            if self.options.online.hardware_assist {
185                log::info!("Applying hardware filters...");
186                let res = self.filter.set_hardware_filter(port);
187                if let Err(error) = res {
188                    log::warn!("Failed to apply some patterns, passing all traffic through Port {}. Reason: {}", port.id, error);
189                };
190            } else {
191                log::info!("No hardware assist configured for port {}, passing all traffic through device.", port.id);
192            }
193        }
194    }
195
196    fn stop_ports(&self) {
197        log::info!("Stopping ports...");
198        for port in self.ports.values() {
199            port.stop();
200        }
201    }
202}

211extern "C" fn launch_rx<S>(arg: *mut c_void) -> i32
212where
213    S: Subscribable,
214{
215    // enforce that workers cores cannot mutate runtime
216    // NICE-TO-HAVE: make this *const and use Mutex for interior mutability
217    let rx_cores = arg as *const BTreeMap<CoreId, RxCore<S>>;
218    let rx_cores = unsafe { &*rx_cores };
219
220    let core_id = CoreId(unsafe { dpdk::rte_lcore_id() } as u32);
221    let rx_core = rx_cores.get(&core_id).expect("Invalid Core");
222
223    // TODO: if the hardware takes care of the whole packet filter, we can skip the packet filter.
224    rx_core.rx_loop();
225    0
