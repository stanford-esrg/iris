17impl FilterParser {
18    /// Parses filter string as a disjunct of `RawPattern`s
19    pub(crate) fn parse_filter(filter_raw: &str) -> Result<Vec<RawPattern>> {
20        let ast = FilterParser::parse_as_ast(filter_raw)?;
21        Ok(FilterParser::flatten_disjunct(ast))
22    }
23
24    fn parse_as_ast(filter_raw: &str) -> Result<Node> {
25        let pairs = FilterParser::parse(Rule::filter, filter_raw);
26        match pairs {
27            Ok(mut pairs) => {
28                let pair = pairs.next().unwrap();
29                FilterParser::parse_disjunct(pair)
30            }
31            Err(_) => bail!(FilterError::InvalidFormat),
32        }
33    }
34
35    // returns a vector of flattened conjuncts (conjunct of predicates)
36    fn flatten_disjunct(disjunct: Node) -> Vec<RawPattern> {
37        let mut flat_conjuncts: Vec<RawPattern> = vec![];
38        if let Node::Disjunct(conjuncts) = disjunct {
39            for conjunct in conjuncts {
40                let mut flat_conjunct = FilterParser::flatten_conjunct(conjunct);
41                flat_conjuncts.append(&mut flat_conjunct)
42            }
43        }
44        flat_conjuncts
45    }
46
47    // returns a vector of RawPatterns
48    fn flatten_conjunct(conjunct: Node) -> Vec<RawPattern> {
49        let mut flat_conjuncts: Vec<RawPattern> = vec![vec![]];
50
51        if let Node::Conjunct(terms) = conjunct {
52            for term in terms {
53                match term {
54                    Node::Predicate(predicate) => {
55                        // append Predicate to end of each flat conjunct
56                        for flat_conj in flat_conjuncts.iter_mut() {
57                            flat_conj.push(predicate.clone());
58                        }
59                    }
60                    Node::Disjunct(disjunct) => {
61                        let flat_disjunct: Vec<RawPattern> =
62                            FilterParser::flatten_disjunct(Node::Disjunct(disjunct));
63                        let cur = flat_conjuncts.clone();
64                        flat_conjuncts.clear();
65                        for conj in flat_disjunct.iter() {
66                            for flat_conj in cur.iter() {
67                                flat_conjuncts.push(FilterParser::combine(flat_conj, conj));
68                            }
69                        }
70                    }
71                    _ => panic!("Conjunct contains non-predicate or disjunct"),
72                }
73            }
74        }
75        flat_conjuncts
76    }

78    fn parse_disjunct(pair: Pair<Rule>) -> Result<Node> {
79        //println!("building from expr: {:#?}", pair);
80        let inner = pair.into_inner();
81        let mut terms = vec![];
82        for pair in inner {
83            if let Rule::sub_expr = pair.as_rule() {
84                terms.push(FilterParser::parse_conjunct(pair)?);
85            }
86        }
87        Ok(Node::Disjunct(terms))
88    }
89
90    fn parse_conjunct(pair: Pair<Rule>) -> Result<Node> {
91        //println!("building from disjunct: {:#?}", pair);
92        let inner = pair.into_inner();
93        let mut terms = vec![];
94        for pair in inner {
95            match pair.as_rule() {
96                Rule::expr => terms.push(FilterParser::parse_disjunct(pair)?),
97                Rule::predicate => terms.extend(FilterParser::parse_predicate(pair)?),
98                _ => (),
99            }
100        }
101        Ok(Node::Conjunct(terms))
102    }
103
104    fn parse_predicate(pair: Pair<Rule>) -> Result<Vec<Node>> {
105        let mut inner = pair.into_inner();
106        let protocol = inner.next().unwrap();
107        match inner.next() {
108            Some(field) => {
109                let op = inner.next().unwrap();
110                let value = inner.next().unwrap();
111
112                match field.as_rule() {
113                    Rule::field => Ok(vec![Node::Predicate(Predicate::Binary {
114                        protocol: FilterParser::parse_protocol(protocol),
115                        field: FilterParser::parse_field(field),
116                        op: FilterParser::parse_binop(op)?,
117                        value: FilterParser::parse_value(value)?,
118                    })]),
119                    Rule::combined_field => {
120                        let mut src_field = "src_".to_owned();
121                        src_field.push_str(field.as_str());
122                        let op = FilterParser::parse_binop(op.clone())?;
123                        let src_node = Node::Predicate(Predicate::Binary {
124                            protocol: FilterParser::parse_protocol(protocol.clone()),
125                            field: FieldName(src_field),
126                            op,
127                            value: FilterParser::parse_value(value.clone())?,
128                        });
129
130                        let mut dst_field = "dst_".to_owned();
131                        dst_field.push_str(field.as_str());
132                        let dst_node = Node::Predicate(Predicate::Binary {
133                            protocol: FilterParser::parse_protocol(protocol.clone()),
134                            field: FieldName(dst_field),
135                            op,
136                            value: FilterParser::parse_value(value.clone())?,
137                        });
138                        match op {
139                            BinOp::Ne => {
140                                // && condition
141                                // e.g., "tcp.port != 80" -> tcp.src_port != 80 and tcp.dst_port != 80"
142                                Ok(vec![src_node, dst_node])
143                            }
144                            _ => {
145                                // || condition
146                                // e.g., "tcp.port = 80" -> tcp.src_port = 80 or tcp.dst_port = 80"
147                                // e.g., "tcp.port > 80" -> tcp.src_port > 80 or tcp.dst_port > 80"
148                                let terms = vec![
149                                    Node::Conjunct(vec![src_node]),
150                                    Node::Conjunct(vec![dst_node]),
151                                ];
152                                Ok(vec![Node::Disjunct(terms)])
153                            }
154                        }
155                    }
156                    _ => bail!(FilterError::InvalidFormat),
157                }
158            }
159            None => {
160                let predicate = FilterParser::parse_protocol(protocol);
161                match predicate.is_supported() {
162                    true => Ok(vec![Node::Predicate(Predicate::Unary {
163                        protocol: predicate,
164                    })]),
165                    false => Ok(vec![Node::Predicate(Predicate::Custom {
166                        name: filterfunc!(predicate.name()),
167                        levels: vec![],        // Temp value; will be filled in later.
168                        matched: true,         // Temp
169                        filtered_data: vec![], // Temp
170                    })]),
171                }
172            }
173        }
174    }

176    fn parse_protocol(pair: Pair<Rule>) -> ProtocolName {

180    fn parse_field(pair: Pair<Rule>) -> FieldName {
181        field!(pair.as_str())

184    fn parse_binop(pair: Pair<Rule>) -> Result<BinOp> {
185        let op_str = pair.as_str().to_string();
186        let mut inner = pair.into_inner();
187        match inner.next().unwrap().as_rule() {
188            Rule::eq_op => Ok(BinOp::Eq),
189            Rule::ne_op => Ok(BinOp::Ne),
190            Rule::ge_op => Ok(BinOp::Ge),
191            Rule::le_op => Ok(BinOp::Le),
192            Rule::gt_op => Ok(BinOp::Gt),
193            Rule::lt_op => Ok(BinOp::Lt),
194            Rule::in_op => Ok(BinOp::In),
195            Rule::re_op => Ok(BinOp::Re),
196            Rule::en_op => Ok(BinOp::En),
197            Rule::byte_re_op => Ok(BinOp::ByteRe),
198            Rule::contains_op => Ok(BinOp::Contains),
199            Rule::not_contains_op => Ok(BinOp::NotContains),
200            _ => bail!(FilterError::InvalidBinOp(op_str)),
201        }
202    }
203
204    fn parse_value(pair: Pair<Rule>) -> Result<Value> {
205        let pair_str = pair.as_str().to_string();
206        let rhs = pair.into_inner().next().unwrap();
207        match rhs.as_rule() {
208            Rule::ipv4_lit => {
209                let ipnet = FilterParser::parse_ipv4(rhs.into_inner())?;
210                Ok(Value::Ipv4(ipnet))
211            }
212            Rule::ipv6_lit => {
213                let ipnet = FilterParser::parse_ipv6(rhs.into_inner())?;
214                Ok(Value::Ipv6(ipnet))
215            }
216            Rule::int_lit => {
217                let val = rhs.as_str().parse::<u64>()?;
218                Ok(Value::Int(val))
219            }
220            Rule::text => {
221                // str_lit is a silent rule, parses directly to Rule::text
222                Ok(Value::Text(rhs.as_str().to_owned()))
223            }
224            Rule::int_range => {
225                let range = FilterParser::parse_int_range(rhs.into_inner())?;
226                Ok(Value::IntRange {
227                    from: *range.start(),
228                    to: *range.end(),
229                })
230            }
231            Rule::byte_lit => {
232                let bytes_as_str = rhs.as_str();
233                let bytes_vec = bytes_as_str
234                    .replace("|", "")
235                    .split_whitespace()
236                    .map(|s| {
237                        u8::from_str_radix(s, 16).unwrap_or_else(|err| {
238                            panic!("Failed to parse {} in {}: {:?}", s, bytes_as_str, err)
239                        })
240                    })
241                    .collect();
242                Ok(Value::Byte(bytes_vec))
243            }
244            _ => bail!(FilterError::InvalidRhsType(pair_str)),
245        }
246    }
247
248    fn parse_ipv4(mut ipv4_lit: Pairs<Rule>) -> Result<Ipv4Net> {
249        let ipv4_addr = ipv4_lit.next().unwrap();
250        let ip = Ipv4Addr::from_str(ipv4_addr.as_str())?;
251        if let Some(ipv4_prefix) = ipv4_lit.next() {
252            let prefix = ipv4_prefix.as_str().parse::<u8>()?;
253            Ok(Ipv4Net::new(ip, prefix)?)
254        } else {
255            Ok(Ipv4Net::new(ip, 32)?)
256        }
257    }
258
259    fn parse_ipv6(mut ipv6_lit: Pairs<Rule>) -> Result<Ipv6Net> {
260        let ipv6_addr = ipv6_lit.next().unwrap();
261        let ip = Ipv6Addr::from_str(ipv6_addr.as_str())?;
262        if let Some(ipv6_prefix) = ipv6_lit.next() {
263            let prefix = ipv6_prefix.as_str().parse::<u8>()?;
264            Ok(Ipv6Net::new(ip, prefix)?)
265        } else {
266            Ok(Ipv6Net::new(ip, 128)?)
267        }
268    }
269
270    fn parse_int_range(mut int_range: Pairs<Rule>) -> Result<RangeInclusive<u64>> {
271        let start = int_range.next().unwrap().as_str().parse::<u64>()?;
272        let end = int_range.next().unwrap().as_str().parse::<u64>()?;
273
274        // disallow same start and end
275        if start >= end {
276            bail!(FilterError::InvalidIntRange { start, end })
277        } else {
278            Ok(RangeInclusive::<u64>::new(start, end))
279        }
280    }

282    // helper function to combine conj1 and conj2, returns new vector of Predicates
283    fn combine(conj1: &[Predicate], conj2: &[Predicate]) -> Vec<Predicate> {
284        let mut result = vec![];
285        for predicate in conj1 {
286            result.push(predicate.clone());
287        }
288        for predicate in conj2 {
289            result.push(predicate.clone())
290        }
291        result
292    }
