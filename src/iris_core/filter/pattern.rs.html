37    // Returns true if self is a fully qualified FlatPattern
38    pub(super) fn is_fully_qualified(&self) -> bool {
39        let (layers, labels) = (&*LAYERS, &*NODE_BIMAP);
40
41        let mut ret = true;
42        let mut prev_header = unwrap_or_ret_false!(labels.get_by_right(&protocol!("ethernet")));
43
44        for predicate in self.predicates.iter() {
45            match predicate {
46                Predicate::Unary { protocol } => {
47                    let cur_header = unwrap_or_ret_false!(labels.get_by_right(protocol));
48                    ret = ret && layers.contains_edge(*cur_header, *prev_header);
49                    prev_header = cur_header;
50                }
51                Predicate::Binary { protocol, .. } => {
52                    let cur_header = unwrap_or_ret_false!(labels.get_by_right(protocol));
53                    ret = ret && (*cur_header == *prev_header)
54                }
55                _ => continue,
56            }
57        }
58        ret
59    }

61    // Returns a vector of fully qualified patterns from self
62    pub(super) fn to_fully_qualified(&self) -> Result<Vec<LayeredPattern>> {
63        if self.is_empty()
64            || (self.predicates.len() == 1
65                && self.predicates[0].get_protocol() == &protocol!("ethernet"))
66        {
67            return Ok(Vec::new());
68        }
69
70        let (layers, labels) = (&*LAYERS, &*NODE_BIMAP);
71
72        let mut node_paths: HashSet<Vec<NodeIndex>> = HashSet::new();
73        let headers = self
74            .predicates
75            .iter()
76            .filter(|c| c.get_protocol() != ProtocolName::none())
77            .map(|c| c.get_protocol())
78            .collect::<HashSet<_>>();
79        for header in headers.iter() {
80            match labels.get_by_right(header) {
81                Some(node) => {
82                    let ethernet = labels
83                        .get_by_right(&protocol!("ethernet"))
84                        .expect("Ethernet not defined.");
85                    let node_path: HashSet<Vec<NodeIndex>> =
86                        algo::all_simple_paths(&layers, *node, *ethernet, 0, None).collect();
87                    node_paths.extend(node_path.iter().map(|p| p.to_vec()));
88                }
89                None => panic!("Predicate header invalid: {}", header),
90            }
91        }
92
93        // all possible fully qualified paths from predicated headers
94        let mut fq_paths = HashSet::new();
95        for node_path in node_paths {
96            let mut fq_path = node_path
97                .iter()
98                .map(|n| labels.get_by_left(n).unwrap().to_owned())
99                .collect::<Vec<_>>();
100            fq_path.remove(fq_path.len() - 1); // remove ethernet
101            fq_path.reverse();
102            fq_paths.insert(fq_path);
103        }
104
105        // build fully qualified patterns (could have multiple per non-fully-qualified pattern)
106        let mut fq_patterns = vec![];
107        for fq_path in fq_paths {
108            let fq_headers: HashSet<&ProtocolName> = fq_path.iter().clone().collect();
109            if headers.is_subset(&fq_headers) {
110                let mut fq_pattern = LayeredPattern::new();
111                for protocol in fq_path.iter() {
112                    let proto_predicates = self
113                        .predicates
114                        .iter()
115                        .filter(|c| c.get_protocol() == protocol && c.is_binary())
116                        .map(|c| c.to_owned())
117                        .collect::<HashSet<_>>();
118
119                    let mut proto_predicates = proto_predicates.into_iter().collect::<Vec<_>>();
120                    proto_predicates.sort();
121
122                    assert!(fq_pattern.add_protocol(protocol.to_owned(), proto_predicates));
123                }
124                fq_patterns.push(fq_pattern);
125            }
126        }
127
128        // Get all custom filters that don't belong to a protocol, validate
129        // that they are valid custom filters, and insert each at the end of
130        // each end-to-end pattern.
131        let custom_preds = self
132            .predicates
133            .iter()
134            .filter(|c| c.is_custom())
135            .map(|c| c.to_owned())
136            .collect::<HashSet<_>>();
137        if !custom_preds.is_empty() {
138            // Add custom filter predicate to end of all patterns
139            for pattern in fq_patterns.iter_mut() {
140                pattern.extend_patterns(&custom_preds);
141            }
142            // ...or as standalone pattern
143            if fq_patterns.is_empty() {
144                fq_patterns.push(LayeredPattern::new());
145                fq_patterns
146                    .last_mut()
147                    .unwrap()
148                    .extend_patterns(&custom_preds);
149            }
150        }
151
152        if fq_patterns.is_empty() {
153            // This happens when the headers provided do not have a directed path to ethernet node
154            // and no custom filters are provided.
155            bail!(FilterError::InvalidPatternLayers(self.to_owned()));
156        }
157        Ok(fq_patterns)
158    }

160    // Validate and populate StateTransitions for the custom predicates
161    // When building a filter from a string, only the name of the custom filter is available.
162    // To populate needed data, we need to correlate these names with the information parsed
163    // at compile-time from the defined custom filters.
164    pub(super) fn handle_custom_predicates(&mut self, valid_preds: &[Predicate]) -> Result<()> {
165        // "Empty" custom predicate with just the name populated
166        for p_empty in &mut self.predicates {
167            if p_empty.is_custom() {
168                let p = valid_preds
169                    .iter()
170                    .find(|p| p.get_name() == p_empty.get_name())
171                    .ok_or(FilterError::InvalidCustomFilter(p_empty.get_name().clone()))?;
172                *p_empty = p.clone();
173                p_empty.set_matched(true); // Set default value
174            }
175        }
176        // empty unary "none" predicates may have been added during parsing
177        self.predicates
178            .retain(|p| !p.is_unary() || p.get_protocol() != ProtocolName::none());
179        Ok(())
180    }

211    // Some custom streaming filters can be `matched` or `continuing`.
212    // We need to distinguish between these at runtime.
213    // Split patterns by streaming filter match state: `matched` vs. `continue`
214    // Note - this could be done more efficiently
215    pub(super) fn split_custom(&self) -> Vec<Self> {
216        if self.predicates.iter().all(|p| !p.is_custom()) {
217            return vec![];
218        }
219        let mut all_patterns: Vec<Self> = vec![Self {
220            predicates: vec![],
221            as_str: self.as_str.clone(),
222        }];
223        for pred in &self.predicates {
224            // For each predicate, either add 1x to each partial or split
225            // patterns by custom streaming/non-custom streaming
226            // This applies to custom filters that either:
227            // - Have multiple separate functions
228            // - Or have any streaming functions
229            if pred.is_custom()
230                && (pred.is_streaming()
231                    || if let Predicate::Custom { levels, .. } = pred {
232                        levels.len() > 1 // REFACTOR: could do pairwise inequality here?
233                    } else {
234                        unreachable!()
235                    })
236            {
237                let mut new_pats = vec![];
238                for partial in &all_patterns {
239                    // Original pattern with matched=true
240                    let mut term_pat = partial.clone();
241                    term_pat.predicates.push(pred.clone());
242
243                    // New pattern with matched=false
244                    let mut nonterm_pat = partial.clone();
245                    let mut pred = pred.clone();
246                    if let Predicate::Custom { matched, .. } = &mut pred {
247                        *matched = false;
248                    }
249                    nonterm_pat.predicates.push(pred);
250
251                    new_pats.push(term_pat);
252                    new_pats.push(nonterm_pat);
253                }
254                all_patterns = new_pats;
255            } else {
256                // Push new predicates
257                for partial in &mut all_patterns {
258                    partial.predicates.push(pred.clone());
259                }
260            }
261        }
262        // Filter out the original pattern: all custom patterns have `matched=true`,
263        // which is covered by the original pattern
264        // That is, retain those for which ANY custom predicate has !matched
265        all_patterns.retain(|pat| {
266            pat.predicates
267                .iter()
268                .any(|pred| pred.is_custom() && pred.is_matching())
269        });
270        all_patterns
271    }

281    /// This layer is the first time that this could match
282    pub(super) fn is_first_match(&self, filter_layer: &StateTransition) -> bool {
283        // Iterate in reverse order since predicates are ordered by layer.
284        // If we hit a predicate that could have been applied at an earlier layer,
285        // going from end to start, we can return early.
286        for pred in self.predicates.iter().rev() {
287            let levels = pred.levels();
288            if pred.is_state() {
289                continue;
290            }
291            for level in &levels {
292                // Predicates are ordered. This indicates that we hit a
293                // predicate that could have been applied at an earlier layer
294                // BEFORE hitting a predicate that must be applied at this layer.
295                if level.compare(filter_layer) == StateTxOrd::Less {
296                    return false;
297                }
298                // Return true if any predicate must be applied at this layer.
299                if level.compare(filter_layer) == StateTxOrd::Equal {
300                    return true;
301                }
302            }
303        }
304        false
305    }

335    pub(super) fn with_l7_state(&self) -> Self {
336        let mut predicates = self.predicates.clone();
337        // LayerState::Discovery
338        // LayerState::Headers
339        // LayerState::Payload
340        if let Some(i) = predicates.iter().position(|x| x.on_proto()) {
341            predicates.insert(
342                i,
343                Predicate::LayerState {
344                    layer: SupportedLayer::L7,
345                    state: LayerState::Headers,
346                    op: BinOp::Ge,
347                },
348            );
349        }
350        if let Some(i) = predicates.iter().position(|x| x.on_session()) {
351            predicates.insert(
352                i,
353                Predicate::LayerState {
354                    layer: SupportedLayer::L7,
355                    state: LayerState::Payload,
356                    op: BinOp::Ge,
357                },
358            );
359        }
360
361        // Move up anything that doesn't rely on the state predicate
362        // REFACTOR: do this recursively
363        if let Some(first_state) = predicates.iter().position(|p| p.is_state()) {
364            if first_state < predicates.len() - 1 {
365                if let Predicate::LayerState { layer, state, .. } = predicates[first_state] {
366                    let back: Vec<_> = predicates.drain(first_state + 1..).collect();
367                    let (pre_state, post_state): (Vec<_>, Vec<_>) =
368                        back.into_iter().partition(|p| !p.depends_on(layer, state));
369                    predicates.splice(first_state..first_state, pre_state);
370                    predicates.extend(post_state);
371                }
372            }
373        }
374        Self {
375            predicates,
376            as_str: self.as_str.clone(),
377        }
378    }

393    // Returns the predicates in FlatPattern that come after (or may come after)
394    // the given StateTransition. REFACTOR: make more efficient.
395    pub(super) fn next_pred(&self, curr: StateTransition) -> Vec<StateTransition> {
396        // All levels that may come after `curr`
397        let mut pat: Vec<_> = self
398            .predicates
399            .iter()
400            .flat_map(|p| p.levels())
401            .filter(|l| matches!(l.compare(&curr), StateTxOrd::Unknown | StateTxOrd::Greater))
402            .collect::<HashSet<_>>() // dedup
403            .into_iter()
404            .collect();
405        // For `matching` streaming filters that are applied at this state,
406        // need to update at future iterations of this filter.
407        if curr.is_streaming()
408            && self
409                .predicates
410                .iter()
411                .any(|p| p.is_custom() && p.is_matching() && p.levels().iter().any(|l| l == &curr))
412        {
413            pat.push(curr);
414        }
415        pat
416    }

418    /// Returns `true` if the patterns are mutually exclusive
419    /// (i.e., there is no way for both patterns to match the same connection).
420    /// We consider two patterns to be mutually exclusive if any predicate in one pattern
421    /// is mutually exclusive with any predicate in the other pattern.
422    #[doc(hidden)]
423    pub fn is_excl(&self, other: &Self) -> bool {
424        for p1 in &self.predicates {
425            for p2 in &other.predicates {
426                if p1 != p2 && p1.is_excl(p2) {
427                    return true;
428                }
429            }
430        }
431        false
432    }

453// Represents a fully qualified pattern, ordered by header layer
454#[derive(Debug, Clone)]
455pub struct LayeredPattern(LinkedHashMap<ProtocolName, Vec<Predicate>>);
456
457impl LayeredPattern {
458    pub(super) fn new() -> Self {
459        LayeredPattern(LinkedHashMap::new())
460    }
461
462    pub(super) fn is_empty(&self) -> bool {
463        self.0.is_empty()
464    }
465
466    // Add `predicates` to the _end_ of the pattern.
467    fn extend_patterns(&mut self, predicates: &HashSet<Predicate>) {
468        if self.0.is_empty() {
469            self.0.insert(
470                ProtocolName::none().clone(),
471                predicates.iter().map(|p| p.to_owned()).collect(),
472            );
473            return;
474        }
475        self.0.insert(
476            ProtocolName::none().clone(),
477            predicates.iter().cloned().collect(),
478        );
479    }
480
481    // Adds predicates on protocol header. Returns true on success
482    fn add_protocol(&mut self, proto_name: ProtocolName, field_predicates: Vec<Predicate>) -> bool {
483        let (layers, labels) = (&*LAYERS, &*NODE_BIMAP);
484
485        // check that there is an edge to previous protocol header
486        // check that field_predicates are all binary
487        // check that field_predicates are all predicates on protocol
488        // ignore custom filters
489        let mut ret = true;
490        let node = unwrap_or_ret_false!(labels.get_by_right(&proto_name));
491        if let Some((outer_proto, _)) = self.0.back() {
492            let prev = unwrap_or_ret_false!(labels.get_by_right(outer_proto));
493            ret = ret && layers.contains_edge(*node, *prev);
494            for pred in field_predicates.iter() {
495                ret = ret
496                    && match pred {
497                        Predicate::Unary { .. } => false,
498                        Predicate::Binary { protocol, .. } => protocol == &proto_name,
499                        Predicate::Custom { .. }
500                        | Predicate::Callback { .. }
501                        | Predicate::LayerState { .. } => false,
502                    }
503            }
504        } else {
505            // IPv4 or IPv6
506            let root = unwrap_or_ret_false!(labels.get_by_right(&protocol!("ethernet")));
507            ret = ret && layers.contains_edge(*node, *root);
508        }
509
510        if ret {
511            self.0.insert(proto_name, field_predicates);
512            ret
513        } else {
514            false
515        }
516    }
517
518    // flattens LayeredPattern to fully qualified FlatPattern
519    pub(super) fn to_flat_pattern(&self) -> FlatPattern {
520        let mut predicates = vec![];
521        for (protocol, field_preds) in self.0.iter() {
522            if protocol != ProtocolName::none() {
523                predicates.push(Predicate::Unary {
524                    protocol: protocol.to_owned(),
525                });
526            }
527            predicates.extend(field_preds.to_owned());
528        }
529        FlatPattern {
530            predicates,
531            as_str: None,
532        }
533    }
534
535    pub(super) fn get_header_predicates(&self) -> &LinkedHashMap<ProtocolName, Vec<Predicate>> {
536        &self.0
537    }
538}

687        let filter = Filter::new(filter_raw, &[]).unwrap();
688        let pattern = filter.get_patterns_flat();

693            .predicates
694            .iter()
695            .all(|x| x.get_protocol() != &protocol!("tls")));
696        let subpattern = pattern.get_subpattern(SupportedLayer::L7, LayerState::Headers);
697        // SNI field check removed
698        assert!(subpattern
699            .predicates
700            .iter()
701            .all(|x| x.get_protocol() != &protocol!("tls") || x.is_unary()));
702    }
