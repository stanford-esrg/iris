12/// The Actions to be tracked for any subset of subscription components
13/// at a filter layer if some filter pattern matches (fully or partially).
14///
15/// The complete set of Actions for a Subscription at a PNode must be built up
16/// by adding datatypes, filter predicates, and callback levels using the
17/// NodeActions (see below). The NodeActions support (1) different actions
18/// for different layer states and (2) APIs for adding the `refresh_at`
19/// components of actions.
20///
21/// **One of these must be built for every filter pattern
22///   (i.e., root-to-node path) due to potential differences in
23///   filter predicate needs.**
24///
25/// NICE-TO-HAVE: ideally we'd use the LayerState predicate instead of having
26/// this weird extra `if_matches` wrapper.
27#[derive(Debug, Clone, Eq, PartialEq, Ord, PartialOrd)]
28pub struct DataActions {
29    /// Optional conditional: these actions are only applied if layer is in
30    /// one of the listed states.
31    pub if_matches: Option<(SupportedLayer, LayerState)>,
32    /// Tracked Actions at the transport layer
33    pub transport: TrackedActions,
34    /// Tracked Actions at encapsulated layers
35    pub layers: [TrackedActions; NUM_LAYERS],
36}
37
38impl std::fmt::Display for DataActions {
39    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
40        if let Some(if_matches) = self.if_matches {
41            write!(f, "{:?}:{:?}-", if_matches.0, if_matches.1)?;
42        }
43        write!(f, "L4:{}; L7:{}", self.transport, self.layers[0])?;
44        Ok(())
45    }
46}

48impl DataActions {
49    pub(crate) fn new() -> Self {
50        DataActions {
51            if_matches: None,
52            transport: TrackedActions::new(),
53            layers: [TrackedActions::new(); NUM_LAYERS],
54        }
55    }
56
57    pub(crate) fn merge(&mut self, peer: &DataActions) {
58        self.transport.extend(&peer.transport);
59        self.layers[0].extend(&peer.layers[0]);
60    }
61
62    pub fn drop(&self) -> bool {
63        self.transport.drop() && self.layers[0].drop()
64    }
65
66    /// Clear the actions that intersect with `peer`
67    pub(crate) fn clear_intersection(&mut self, peer: &DataActions) {
68        if self.if_matches != peer.if_matches {
69            return;
70        }
71        self.transport.clear_intersection(&peer.transport);
72        self.layers[0].clear_intersection(&peer.layers[0]);
73    }
74
75    // Get the actions associated with a matching `pred`.
76    pub(crate) fn from_stream_pred(
77        pred: &Predicate,
78        next_preds: Vec<StateTransition>,
79        filter_layer: StateTransition,
80        curr_state_pred: &Option<Predicate>,
81    ) -> DataActions {
82        if let Predicate::Custom {
83            name,
84            levels,
85            matched,
86            filtered_data: _,
87        } = pred
88        {
89            assert!(!*matched);
90            let spec = StateTransitionSpec {
91                updates: levels.iter().flatten().cloned().collect(),
92                name: name.clone().0,
93            };
94            let actions = spec.to_actions(filter_layer);
95            let curr_state = match curr_state_pred {
96                Some(p) => match p {
97                    Predicate::LayerState { layer, state, .. } => Some((*layer, *state)),
98                    _ => panic!("State predicate is {}", p),
99                },
100                None => None,
101            };
102            for mut a in actions.actions {
103                if a.if_matches == curr_state {
104                    for p in next_preds {
105                        a.transport.refresh_at[p.as_usize()] |= a.transport.active;
106                        a.layers[0].refresh_at[p.as_usize()] |= a.layers[0].active;
107                    }
108                    return a;
109                }
110            }
111            return DataActions::new();
112        }
113        panic!("From_stream_pred called on {:?}", pred);
114    }
115
116    pub(crate) fn refresh_at(&self) -> Vec<StateTransition> {
117        let mut ret: Vec<StateTransition> = (0..NUM_STATE_TRANSITIONS)
118            .filter(|&i| {
119                self.transport.refresh_at[i] != 0
120                    || self.layers.iter().any(|layer| layer.refresh_at[i] != 0)
121            })
122            .map(StateTransition::from_usize)
123            .collect();
124        ret.sort();
125        ret.dedup();
126        ret
127    }
128}

130/// Actions for a single subscription that will be stored at a node in a PTree.
131/// This must be built up by first adding `datatypes` and custom filters and then
132/// adding `next predicates` or `streaming callbacks` to determine when the node's
133/// actions could potentially be updated.
134#[derive(Debug, Clone, Eq, PartialEq)]
135pub struct NodeActions {
136    /// Essentially, there will be multiple elements here if there are different
137    /// branches depending on some layer's state.
138    pub actions: Vec<DataActions>,
139    /// Set to `true` the first time a filter predicate or streaming callback level
140    /// is added to ensure that no further datatypes are added afterwards.
141    pub(crate) end_datatypes: bool,
142    /// The state transition PTree that this Node is part of.
143    pub(crate) filter_layer: StateTransition,
144}
145
146impl std::fmt::Display for NodeActions {
147    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
148        for (i, d) in self.actions.iter().enumerate() {
149            if i > 0 {
150                write!(f, " ")?; // spacing
151            }
152            write!(f, "{}", d)?;
153        }
154        Ok(())
155    }
156}

158impl NodeActions {
159    pub(crate) fn new(filter_layer: StateTransition) -> Self {
160        Self {
161            actions: vec![],
162            end_datatypes: false,
163            filter_layer,
164        }
165    }
166
167    /// Add `new` to `actions` by either:
168    /// - Appending it, or
169    /// - Updating an existing action that has the same preconditions
170    pub(crate) fn push_action(&mut self, new: DataActions) {
171        let curr = self
172            .actions
173            .iter_mut()
174            .find(|a| a.if_matches == new.if_matches);
175        match curr {
176            Some(curr) => {
177                curr.transport.extend(&new.transport);
178                curr.layers[0].extend(&new.layers[0]);
179            }
180            None => self.actions.push(new),
181        }
182    }
183
184    /// Update actions with an additional datatype.
185    pub(crate) fn add_datatype(&mut self, spec: &StateTransitionSpec) {
186        assert!(
187            !self.end_datatypes,
188            "Cannot add new datatypes after adding filter predicates."
189        );
190        let actions = spec.to_actions(self.filter_layer);
191        for a in actions.actions {
192            self.push_action(a);
193        }
194    }
195
196    /// Must be added after full NodeActions for a specific subscription has been built up,
197    /// including the datatypes required for filters.
198    /// Updates Actions to be "refreshed" at the next layer where new relevant information
199    /// might be available (i.e., next filter predicate can be evaluated).
200    pub(crate) fn push_filter_pred(&mut self, next_pred: &StateTransition) {
201        self.end_datatypes = true;
202        for a in self.actions.iter_mut() {
203            a.transport.refresh_at[next_pred.as_usize()] |= a.transport.active;
204            a.layers[0].refresh_at[next_pred.as_usize()] |= a.layers[0].active;
205        }
206    }
207
208    /// Manually indicate that we're done adding new data
209    pub(crate) fn end_datatypes(&mut self) {
210        self.end_datatypes = true;
211    }
212
213    /// Must be added after full subscription spec has been built up with all datatypes,
214    /// including the datatypes required for filters.
215    /// `level` should indicate the level at which the callback could unsubscribe.
216    /// Updates subscription Actions to be "refreshed" at this point.
217    pub(crate) fn push_cb(&mut self, level: StateTransition) {
218        self.push_filter_pred(&level);
219    }
220
221    /// Merge two NodeActions together.
222    /// This is typically needed when a Node already has actions
223    /// accumulated and another subscription (sub-)pattern terminates
224    /// at the node.
225    #[allow(dead_code)]
226    pub(crate) fn merge(&mut self, peer: &NodeActions) {
227        assert!(
228            self.end_datatypes && peer.end_datatypes || self.actions.is_empty(),
229            "SELF: {}\nPEER: {}",
230            self,
231            peer
232        );
233        assert!(self.filter_layer == peer.filter_layer);
234        for a in &peer.actions {
235            self.push_action(a.clone());
236        }
237        self.end_datatypes = true;
238    }
239
240    /// Returns `true` if no actions
241    #[allow(dead_code)]
242    pub(crate) fn drop(&self) -> bool {
243        self.actions.is_empty() || self.actions.iter().all(|a| a.transport.drop())
244    }
245}

247/// Compile-time struct for representing a datatype required for a callback
248/// or custom filter predicate.
249/// Might also be used to represent a stateful custom filter predicate.
250#[derive(Debug, Clone, PartialEq, Eq)]
251pub struct StateTransitionSpec {
252    /// Updates: streaming updates and state transitions requested.
253    pub updates: Vec<StateTransition>,
254    /// The name of the datatype as a string
255    pub name: String,
256}
257
258impl StateTransitionSpec {
259    /// From a filter predicate
260    pub(crate) fn from_pred(pred: &Predicate) -> Option<Self> {
261        // Predicates that have already matched don't need add'l actions
262        if pred.is_custom() && !pred.is_matching() {
263            return None;
264        }
265        Some(Self {
266            updates: pred.levels().clone(),
267            name: format!("{}", pred),
268        })
269    }
270
271    /// For a given filter layer (state transition), return the actions that the
272    /// datatype requires.
273    ///
274    /// Example: A TLS handshake requires a "headers parsed" state TX. We must generate
275    /// the actions that will allow that state transition to execute. The
276    /// framework must (1) 'pass through' at the transport layer (to L7) and (2) parse
277    /// at L7, and it must continue doing this _until_ the L7 headers are parsed or
278    /// we fail to identify the protocol.
279    /// If the filter layer is guaranteed to execute before the end
280    /// of the L7 headers, then we can proceed with these actions. If the layer is
281    /// guaranteed to execute after the end of the L7 headers, we should do nothing.
282    /// If the layers are not comparable (may happen in any order), then the actions
283    /// taken depend on the L7 state.
284    pub(crate) fn to_actions(&self, filter_layer: StateTransition) -> NodeActions {
285        let mut actions = NodeActions::new(filter_layer);
286        if filter_layer == StateTransition::L4Terminated {
287            return actions;
288        }
289        let l7_idx = SupportedLayer::L7 as usize - 1;
290        for level in &self.updates {
291            let cmp = filter_layer.compare(level);
292            let mut a = DataActions::new();
293            match level {
294                StateTransition::L4FirstPacket => {
295                    // Nothing required.
296                    // Datatype can be delivered or cached on first packet filter.
297                    continue;
298                }
299                StateTransition::L4EndHshk => {
300                    match cmp {
301                        StateTxOrd::Less | StateTxOrd::Unknown => {
302                            // Detecting the handshake requires invoking reassembly
303                            a.transport.active |= Actions::Parse;
304                            // ...until the handshake has been parsed
305                            a.transport.refresh_at[level.as_usize()] |= Actions::Parse;
306                            // Only if pre-handshake
307                            if matches!(cmp, StateTxOrd::Unknown) {
308                                a.if_matches = Some((SupportedLayer::L4, LayerState::Headers));
309                            }
310                            actions.push_action(a);
311                        }
312                        _ => continue,
313                    }
314                }
315                StateTransition::InL4Conn => {
316                    // "In" suggests an update is required.
317                    // By itself, an L4 "In" datatype doesn't "unsubscribe".
318                    a.transport.active |= Actions::Update;
319                    actions.push_action(a);
320                }
321                StateTransition::InL4Stream => {
322                    // For L4, `Actions::Parse` indicates update post-reassembly
323                    a.transport.active |= Actions::Parse;
324                    actions.push_action(a);
325                }
326                StateTransition::L7OnDisc => {
327                    match cmp {
328                        StateTxOrd::Equal | StateTxOrd::Greater | StateTxOrd::Any => continue,
329                        StateTxOrd::Less | StateTxOrd::Unknown => {
330                            // Must pass to subsequent layer.
331                            // Note this will implicitly reassemble.
332                            a.transport.active |= Actions::PassThrough;
333                            a.transport.refresh_at[level.as_usize()] |= Actions::PassThrough;
334                            // "Parse" at L7 in order to identify protocol, until protocol identified
335                            // (or all protocols ruled out).
336                            a.layers[l7_idx].active |= Actions::Parse;
337                            a.layers[l7_idx].refresh_at[level.as_usize()] |= Actions::Parse;
338                            if matches!(cmp, StateTxOrd::Unknown) {
339                                a.if_matches = Some((SupportedLayer::L7, LayerState::Discovery));
340                            }
341                            actions.push_action(a);
342                        }
343                    }
344                }
345                StateTransition::L7EndHdrs => match cmp {
346                    StateTxOrd::Equal | StateTxOrd::Greater | StateTxOrd::Any => continue,
347                    StateTxOrd::Less | StateTxOrd::Unknown => {
348                        a.transport.active |= Actions::PassThrough;
349                        a.transport.refresh_at[level.as_usize()] |= Actions::PassThrough;
350                        a.layers[l7_idx].active |= Actions::Parse;
351                        a.layers[l7_idx].refresh_at[level.as_usize()] |= Actions::Parse;
352                        if matches!(cmp, StateTxOrd::Unknown) {
353                            // Differentiate between L7 Disc, Headers
354                            a.if_matches = Some((SupportedLayer::L7, LayerState::Discovery));
355                            actions.push_action(a.clone());
356                            a.if_matches = Some((SupportedLayer::L7, LayerState::Headers));
357                        }
358                        actions.push_action(a);
359                    }
360                },
361                StateTransition::L4Terminated => {
362                    let mut a = DataActions::new();
363                    a.transport.active |= Actions::Track;
364                    actions.push_action(a);
365                }
366                // No actions
367                StateTransition::Packet => continue,
368            }
369        }
370        actions
371    }
372}

374/// Structure representing a single callback function.
375///
376/// If a subscription specifies multiple callback functions (i.e.,
377/// on a struct), one of these must be created for each function.
378#[derive(Debug, Clone, Eq, PartialEq)]
379pub struct CallbackSpec {
380    /// If the callback explicitly specifies when to be invoked
381    pub expl_level: Option<StateTransition>,
382    /// Datatype inputs to the callback
383    pub datatypes: Vec<StateTransitionSpec>,
384    /// This callback cannot be optimized out.
385    /// Typically true if ``FilterStr`` (i.e., information
386    /// about the specific filter matched) is a parameter.
387    pub must_deliver: bool,
388    /// The framework must track when it has invoked this callback
389    /// in order to ensure it is invoked at most once.
390    /// Typically required if the associated filter includes a
391    /// streaming predicate but the callback is stateless
392    /// and not streaming.
393    pub invoke_once: bool,
394    /// The name of the callback function (excluding parameters).
395    /// Also excludes the struct for stateful callbacks, if applicable.
396    pub as_str: String,
397    /// Subscription string representation
398    /// This should be the name of the callback struct (for stateful CBs)
399    /// or the name of the callback function for ungrouped CBs.
400    pub subscription_id: String,
401    /// "Expensive" tracked datatypes (by name)
402    /// Used to indicate that a tree should track the match state of the
403    /// datatype in order to `clear` it if it goes out of scope.
404    /// This may limit filter optimizations.
405    pub filtered_data: Vec<String>,
406}
407
408impl Hash for CallbackSpec {
409    fn hash<H: Hasher>(&self, state: &mut H) {
410        self.as_str.hash(state);
411        self.subscription_id.hash(state);
412    }
413}
414
415impl CallbackSpec {
416    pub(super) fn get_datatypes(&self) -> Vec<StateTransitionSpec> {
417        let mut datatypes: Vec<_> = self.datatypes.to_vec();
418        if let Some(expl_level) = self.expl_level {
419            // Requires streaming `updates` or the level cannot be
420            // inferred from the datatype alone
421            datatypes.push(StateTransitionSpec {
422                updates: vec![expl_level],
423                name: self.as_str.clone(),
424            });
425        }
426        datatypes
427    }
428
429    pub fn is_streaming(&self) -> bool {
430        match self.expl_level {
431            Some(l) => l.is_streaming(),
432            None => false,
433        }
434    }
435
436    pub fn is_grouped(&self) -> bool {
437        self.as_str != self.subscription_id
438    }
439}

442/// Utility for tracking, for a subscription pattern*, the Levels needed to
443/// correctly apply (1) all filter predicates, (2) all datatype updates,
444/// and (3), if applicable, updates to a streaming callback.
445///
446/// *For each subscription, one of these structs is needed for each end-to-end
447/// filter pattern (i.e., root-to-leaf predicate tree path).
448#[derive(Debug, Clone)]
449pub struct SubscriptionLevel {
450    /// Levels at which datatype updates need to happen
451    pub datatypes: HashSet<StateTransition>,
452    /// Levels at which a new filter predicate needs to be applied
453    pub filter_preds: HashSet<StateTransition>,
454    /// If the callback explicitly requests to be delivered at a level
455    pub callback: Option<StateTransition>,
456}
457
458impl SubscriptionLevel {
459    pub fn new(
460        data: &Vec<StateTransitionSpec>,
461        preds: &FlatPattern,
462        cb: Option<StateTransition>,
463    ) -> Self {
464        let mut ret = Self::empty();
465        for d in data {
466            ret.add_datatype(d);
467        }
468        for p in &preds.predicates {
469            ret.add_filter_pred(&p.levels());
470        }
471        ret.add_callback(cb);
472        ret
473    }
474
475    /// Can the callback be invoked or CB timer started at `curr` state tx?
476    /// Delivery can kick off if the current state transition is
477    /// _not less than_ any of the predicates and it is _equal_ to at least one
478    /// of the predicates (i.e., this may be the first state TX where the "not less than"
479    /// bound is true). This must be true for all filter and datatype predicates.
480    pub fn can_deliver(&self, curr: &StateTransition) -> bool {
481        // Only deliver callback at explicitly-specified level.
482        // Note that the connection tracker invokes `update` AFTER
483        // parsing and applying state transitions. For example, an `InL4Conn`
484        // update will be invoked after an `L7OnDisc` state transition, if the
485        // same PDU triggers both.
486        if let Some(expl_level) = &self.callback {
487            // E.g., L4OnTerminated, L4FirstPacket
488            if !expl_level.is_streaming() {
489                return curr == expl_level;
490            }
491            // Streaming callbacks will be delivered in `update` and only
492            // need to be delivered here if this is the first time they CAN
493            // be delivered (the corresponding `update` would have already passed).
494        }
495        // Create iterator over all filter and datatype predicates
496        let mut iter = self.datatypes.iter().chain(self.filter_preds.iter());
497        iter.clone()
498            .all(|l| !matches!(curr.compare(l), StateTxOrd::Less))
499            && iter.any(|l| curr.compare(l) == StateTxOrd::Equal)
500    }
501
502    /// Can the pattern of predicates be skipped at this state transition layer?
503    /// That is, is this subscription guaranteed to have terminated at a level
504    /// that is strictly less than (before) the current level?
505    pub fn can_skip(&self, curr: &StateTransition) -> bool {
506        let cb_done = match self.callback {
507            Some(cb) => curr.compare(&cb) == StateTxOrd::Greater,
508            None => false,
509        };
510        // All datatype and filter predicate levels are strictly less than `curr`
511        cb_done
512            && self
513                .datatypes
514                .iter()
515                .chain(self.filter_preds.iter())
516                .all(|l| curr.compare(l) == StateTxOrd::Greater)
517    }
518
519    // -- utilities for iteratively building up a spec --
520
521    pub(crate) fn empty() -> Self {
522        Self {
523            datatypes: HashSet::new(),
524            filter_preds: HashSet::new(),
525            callback: None,
526        }
527    }
528
529    /// Add the Level of the streaming callback (e.g., "In L4 payload")
530    pub(crate) fn add_callback(&mut self, level: Option<StateTransition>) {
531        self.callback = level;
532    }
533
534    /// Add a datatype requested in a callback
535    pub(crate) fn add_datatype(&mut self, data: &StateTransitionSpec) {
536        for d in &data.updates {
537            self.datatypes.insert(*d);
538        }
539    }
540
541    /// Add a filter predicate
542    pub(crate) fn add_filter_pred(&mut self, pred: &Vec<StateTransition>) {
543        for d in pred {
544            self.filter_preds.insert(*d);
545        }
546    }
547}

549#[derive(Debug, Clone, Eq, PartialEq)]
550pub struct FilteredDatatype {
551    pub name: String,
552    pub refresh_at: Vec<StateTransition>,
553}
554
555impl std::fmt::Display for FilteredDatatype {
556    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
557        if self.refresh_at.is_empty() {
558            return write!(f, "{}", self.name);
559        }
560        let state_tx_list = self
561            .refresh_at
562            .iter()
563            .map(|tx| tx.to_string())
564            .collect::<Vec<_>>()
565            .join(",");
566        write!(f, "{}->({})", self.name, state_tx_list)
567    }
568}
569
570impl FilteredDatatype {
571    pub fn new(name: String, mut refresh_at: Vec<StateTransition>) -> Self {
572        refresh_at.sort_by_key(|tx| tx.as_usize());
573        refresh_at.dedup();
574        Self { name, refresh_at }
575    }
576
577    pub fn extend(&mut self, other: &FilteredDatatype) {
578        assert!(self.name == other.name);
579        self.refresh_at.extend(other.refresh_at.iter().cloned());
580        self.refresh_at.sort_by_key(|tx| tx.as_usize());
581        self.refresh_at.dedup();
582    }
583
584    pub fn refresh_as_u8(&self) -> u8 {
585        let mut ret = 0;
586        for tx in &self.refresh_at {
587            ret |= 1 << tx.as_usize();
588        }
589        ret
590    }
591
592    pub fn concat(
593        l: &HashMap<String, FilteredDatatype>,
594        r: &HashMap<String, FilteredDatatype>,
595    ) -> HashMap<String, FilteredDatatype> {
596        let mut ret: HashMap<String, FilteredDatatype> = HashMap::new();
597        for (k, v) in l.iter().chain(r.iter()) {
598            if let Some(existing) = ret.get_mut(k) {
599                existing.extend(v);
600            } else {
601                ret.insert(k.clone(), v.clone());
602            }
603        }
604        ret
605    }
606
607    pub fn is_contained_in(&self, other: &FilteredDatatype) -> bool {
608        assert!(self.name == other.name);
609        self.refresh_at
610            .iter()
611            .all(|tx| other.refresh_at.contains(tx))
612    }
613}
614
615impl Hash for FilteredDatatype {
616    fn hash<H: Hasher>(&self, state: &mut H) {
617        assert!(self.refresh_at.is_sorted());
618        self.name.hash(state);
619        self.refresh_at.hash(state);
620    }
621}
