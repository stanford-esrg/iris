66    pub(super) fn from_layered_pattern(pattern: &LayeredPattern) -> Result<FlowPattern> {
67        let mut flow_pattern = FlowPattern { items: Vec::new() };
68
69        for (protocol, field_preds) in pattern.get_header_predicates().iter() {
70            match protocol.name() {
71                "ipv4" => flow_pattern.append_ipv4(field_preds)?,
72                "ipv6" => flow_pattern.append_ipv6(field_preds)?,
73                "tcp" => flow_pattern.append_tcp(field_preds)?,
74                "udp" => flow_pattern.append_udp(field_preds)?,
75                _ => bail!(FilterError::InvalidHeader(protocol.name().to_owned())),
76            }
77        }
78        Ok(flow_pattern)
79    }

81    pub(super) fn append_ipv4(&mut self, predicates: &[Predicate]) -> Result<()> {
82        let mut ipv4_spec: dpdk::rte_flow_item_ipv4 = unsafe { mem::zeroed() };
83        let mut ipv4_mask: dpdk::rte_flow_item_ipv4 = unsafe { mem::zeroed() };
84
85        for pred in predicates.iter() {
86            match pred {
87                Predicate::Custom { .. } => {
88                    bail!(FilterError::InvalidPredType("custom".to_owned()))
89                }
90                Predicate::Callback { .. } => {
91                    bail!(FilterError::InvalidPredType("callback".to_owned()))
92                }
93                Predicate::LayerState { .. } => {
94                    bail!(FilterError::InvalidPredType("layerstate".to_owned()))
95                }
96                Predicate::Unary { .. } => bail!(FilterError::InvalidPredType("unary".to_owned())),
97                Predicate::Binary {
98                    protocol: _,
99                    field,
100                    op: _,
101                    value,
102                } => match field.name() {
103                    "version_ihl" => match value {
104                        Value::Int(i) => {
105                            if let Ok(val) = u8::try_from(*i) {
106                                ipv4_spec.hdr.set_version_ihl(val);
107                                ipv4_mask.hdr.set_version_ihl(u8::MAX);
108                            } else {
109                                bail!(FilterError::InvalidRhsValue(value.to_string()))
110                            }
111                        }
112                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
113                    },
114                    "type_of_service" => match value {
115                        Value::Int(i) => {
116                            if let Ok(val) = u8::try_from(*i) {
117                                ipv4_spec.hdr.type_of_service = val;
118                                ipv4_mask.hdr.type_of_service = u8::MAX;
119                            } else {
120                                bail!(FilterError::InvalidRhsValue(value.to_string()))
121                            }
122                        }
123                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
124                    },
125                    "total_length" => match value {
126                        Value::Int(i) => {
127                            if let Ok(val) = u16::try_from(*i) {
128                                ipv4_spec.hdr.total_length = val.to_be();
129                                ipv4_mask.hdr.total_length = u16::MAX;
130                            } else {
131                                bail!(FilterError::InvalidRhsValue(value.to_string()))
132                            }
133                        }
134                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
135                    },
136                    "identification" => match value {
137                        Value::Int(i) => {
138                            if let Ok(val) = u16::try_from(*i) {
139                                ipv4_spec.hdr.packet_id = val.to_be();
140                                ipv4_mask.hdr.packet_id = u16::MAX;
141                            } else {
142                                bail!(FilterError::InvalidRhsValue(value.to_string()))
143                            }
144                        }
145                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
146                    },
147                    "flags_to_fragment_offset" => match value {
148                        Value::Int(i) => {
149                            if let Ok(val) = u16::try_from(*i) {
150                                ipv4_spec.hdr.fragment_offset = val.to_be();
151                                ipv4_mask.hdr.fragment_offset = u16::MAX;
152                            } else {
153                                bail!(FilterError::InvalidRhsValue(value.to_string()))
154                            }
155                        }
156                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
157                    },
158                    "time_to_live" => match value {
159                        Value::Int(i) => {
160                            if let Ok(val) = u8::try_from(*i) {
161                                ipv4_spec.hdr.time_to_live = val;
162                                ipv4_mask.hdr.time_to_live = u8::MAX;
163                            } else {
164                                bail!(FilterError::InvalidRhsValue(value.to_string()))
165                            }
166                        }
167                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
168                    },
169                    "protocol" => match value {
170                        Value::Int(i) => {
171                            if let Ok(val) = u8::try_from(*i) {
172                                ipv4_spec.hdr.next_proto_id = val;
173                                ipv4_mask.hdr.next_proto_id = u8::MAX;
174                            } else {
175                                bail!(FilterError::InvalidRhsValue(value.to_string()))
176                            }
177                        }
178                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
179                    },
180                    "header_checksum" => match value {
181                        Value::Int(i) => {
182                            if let Ok(val) = u16::try_from(*i) {
183                                ipv4_spec.hdr.hdr_checksum = val.to_be();
184                                ipv4_mask.hdr.hdr_checksum = u16::MAX;
185                            } else {
186                                bail!(FilterError::InvalidRhsValue(value.to_string()))
187                            }
188                        }
189                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
190                    },
191                    "src_addr" => match value {
192                        Value::Ipv4(ipv4net) => {
193                            ipv4_spec.hdr.src_addr = u32::from(ipv4net.addr()).to_be();
194                            ipv4_mask.hdr.src_addr = u32::from(ipv4net.netmask()).to_be();
195                        }
196                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
197                    },
198                    "dst_addr" => match value {
199                        Value::Ipv4(ipv4net) => {
200                            ipv4_spec.hdr.dst_addr = u32::from(ipv4net.addr()).to_be();
201                            ipv4_mask.hdr.dst_addr = u32::from(ipv4net.netmask()).to_be();
202                        }
203                        _ => bail!(FilterError::InvalidRhsType(value.to_string())),
204                    },
205                    _ => bail!(FilterError::InvalidField(field.name().to_owned())),
206                },
207            }
208        }
209
210        let ipv4_item = FlowItem::<dpdk::rte_flow_item_ipv4> {
211            item_type: dpdk::rte_flow_item_type_RTE_FLOW_ITEM_TYPE_IPV4,
212            spec: ipv4_spec,
213            mask: ipv4_mask,
214        };
215        self.items.push(Box::new(ipv4_item));
216        Ok(())
217    }
