15// A node representing a predicate in the tree
16#[derive(Debug, Clone)]
17pub struct PNode {
18    // Predicate represented by this PNode
19    pub pred: Predicate,
20
21    // Child PNodes
22    pub children: Vec<PNode>,
23
24    // Predicate is mutually exclusive with the
25    // predicate in the node preceding it in child list
26    pub if_else: bool,
27
28    // `Actions` struct for this node.
29    pub actions: DataActions,
30
31    // Subscriptions that can be invoked at this node.
32    // That is, all for which `can_deliver` on its `SubscriptionLevel``
33    // returned true.
34    pub deliver: HashSet<CallbackSpec>,
35
36    // Callback groups that should be marked "active" (filter matched).
37    pub matched: HashSet<String>,
38
39    // Datatypes that remain "in scope" at this node
40    // Only tracked for "expensive" datatypes
41    pub filtered_datatypes: HashMap<String, FilteredDatatype>,
42
43    // Identifier
44    pub id: usize,
45
46    // String representation of the filter that this node terminates
47    pub filter_str: Option<String>,
48}

50impl PNode {
51    fn new(pred: Predicate, id: usize) -> Self {
52        PNode {
53            pred,
54            children: vec![],
55            if_else: false,
56            actions: DataActions::new(),
57            deliver: HashSet::new(),
58            matched: HashSet::new(),
59            filtered_datatypes: HashMap::new(),
60            id,
61            filter_str: None,
62        }
63    }

65    /// -- Utilities for comparing predicates when inserting nodes -- //
66    // Utility to check whether a descendant exists
67    // Helper for `get_descendant`, which must be invoked in an `if` block
68    // due to borrow checker
69    fn has_descendant(&self, pred: &Predicate, state: &Option<Predicate>) -> bool {
70        if &self.pred == pred {
71            return true;
72        }
73        for n in &self.children {
74            if n.pred.is_state() && Some(&n.pred) != state.as_ref() {
75                return false;
76            }
77            if n.pred.is_callback() && &n.pred != pred {
78                // Stop at callbacks
79                return false;
80            }
81            if &n.pred == pred {
82                return true;
83            }
84            if pred.is_child(&n.pred) && n.has_descendant(pred, state) {
85                return true;
86            }
87        }
88        false
89    }
90
91    // See above
92    fn get_descendant(&mut self, pred: &Predicate) -> Option<&mut PNode> {
93        if &self.pred == pred {
94            return Some(self);
95        }
96        for n in &mut self.children {
97            // found exact match
98            if &n.pred == pred {
99                return Some(n);
100            }
101            // node is a parent - keep descending
102            if pred.is_child(&n.pred) {
103                if let Some(c) = n.get_descendant(pred) {
104                    return Some(c);
105                }
106            }
107        }
108        None
109    }
110
111    // Returns true if `self` has `pred` as a direct child
112    fn has_child(&self, pred: &Predicate) -> bool {
113        self.children.iter().any(|n| &n.pred == pred)
114    }
115
116    // See above
117    fn get_child(&mut self, pred: &Predicate) -> &mut PNode {
118        self.children.iter_mut().find(|n| &n.pred == pred).unwrap()
119    }
120
121    // True if `self` has children that should be (more specific)
122    // children of `pred`
123    fn has_children_of(&self, pred: &Predicate) -> bool {
124        self.children.iter().any(|n| n.pred.is_child(pred))
125    }
126
127    // Returns all of the PNodes that should be children of `pred`,
128    // while retaining in `self.children` the nodes that are
129    // not children of `pred`
130    fn get_children_of(&mut self, pred: &Predicate) -> Vec<PNode> {
131        let new;
132        let children = std::mem::take(&mut self.children);
133
134        (new, self.children) = children.into_iter().partition(|p| p.pred.is_child(pred));
135
136        new
137    }
138
139    // Returns a reference to a PNode that is a child of `self`
140    // that can act as "parent" of `pred`.
141    fn get_parent_candidate(&mut self, pred: &Predicate) -> Option<&mut PNode> {
142        self.children.iter_mut().find(|n| pred.is_child(&n.pred))
143    }
144
145    // Returns true if (1) both `self` and `peer` have equal node-to-leaf paths
146    // and (2) actions/CB/tracked datatypes are the same.
147    // This is useful for marking nodes as mutually exclusive even
148    // if there predicates are not mutually exclusive.
149    fn outcome_eq(&self, peer: &PNode) -> bool {
150        if self != peer {
151            return false;
152        }
153        (self.children.is_empty() && peer.children.is_empty()) || self.all_paths_eq(peer)
154    }
155
156    // True if there is a PNode that can act as parent of `pred`.
157    fn has_parent(&self, pred: &Predicate) -> bool {
158        for n in &self.children {
159            if pred.is_child(&n.pred) {
160                return true;
161            }
162        }
163        false
164    }
165
166    // Returns a node that can act as a parent to `pred`, or None.
167    // The most "narrow" parent condition will be returned if multiple exist.
168    fn get_parent(
169        &mut self,
170        pred: &Predicate,
171        tree_size: usize,
172        state: &Option<Predicate>,
173    ) -> Option<&mut PNode> {
174        // This is messy, but directly iterating through children or
175        // recursing will raise flags with the borrow checker.
176        let mut node = self;
177        for _ in 0..tree_size {
178            // Stop traversing if hit a state dependency
179            if node.pred.is_state() && (state.is_none() || Some(&node.pred) != state.as_ref()) {
180                return None;
181            }
182            // Checked for `Some` on last iteration
183            let next = node.get_parent_candidate(pred)?;
184            if next.get_parent_candidate(pred).is_none() {
185                // `next` is the last possible parent at this stage
186                return Some(next);
187            } else {
188                // There are more potential parents
189                node = next;
190            }
191        }
192        None
193    }
194
195    // Returns `true` if a condition cannot be removed from the filter due to
196    // its role extracting data needed for a subsequent condition.
197    // For example, getting `ipv4` is necessary for checking `ipv4.src_addr`.
198    fn extracts_protocol(&self, filter_layer: StateTransition) -> bool {
199        // Filters that parse raw packets are special case
200        // Need upper layers to extract inner from mbuf
201        // E.g.: need ipv4 header to parse tcp
202        if matches!(filter_layer, StateTransition::L4FirstPacket)
203            && self.pred.is_unary()
204            && self.children.iter().any(|n| n.pred.is_unary())
205        {
206            return true;
207        }
208        self.pred.is_unary()
209            && self
210                .children
211                .iter()
212                .any(|n| self.pred.get_protocol() == n.pred.get_protocol() && n.pred.is_binary())
213    }
214
215    fn has_op(&self) -> bool {
216        !self.actions.drop()
217            || !self.deliver.is_empty()
218            || !self.matched.is_empty()
219            || !self.filtered_datatypes.is_empty()
220    }
221
222    // Populates `paths` with all root-to-leaf paths originating
223    // at node `self`.
224    fn get_paths(&self, curr_path: &mut Vec<String>, paths: &mut Vec<String>) {
225        if self.children.is_empty() && !curr_path.is_empty() {
226            paths.push(curr_path.join(","));
227        } else {
228            for c in &self.children {
229                curr_path.push(format!("{}", c));
230                c.get_paths(curr_path, paths);
231            }
232        }
233        curr_path.pop();
234    }
235
236    // Returns true if all root-to-leaf paths originating at `self`
237    // are the same as all root-to-leaf paths originating at `other`.
238    // This is applied to determine if two nodes have the same outcome
239    // (subsequent conditions, actions, delivery) if predicate is `true`.
240    fn all_paths_eq(&self, other: &PNode) -> bool {
241        if self.children.is_empty() && other.children.is_empty() {
242            return true;
243        }
244        let mut paths = vec![];
245        let mut curr = vec![];
246        self.get_paths(&mut curr, &mut paths);
247        let mut peer_paths = vec![];
248        curr = vec![];
249        other.get_paths(&mut curr, &mut peer_paths);
250        peer_paths == paths
251    }
252
253    pub(super) fn insert_filtered_data(&mut self, data: FilteredDatatype) {
254        if let Some(dt) = self.filtered_datatypes.get_mut(&data.name) {
255            dt.extend(&data);
256        } else {
257            self.filtered_datatypes.insert(data.name.clone(), data);
258        }
259    }
260}

262impl fmt::Display for PNode {
263    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
264        let mut invoke: Vec<&str> = self.deliver.iter().map(|d| d.as_str.as_str()).collect();
265        invoke.sort_unstable();
266        let invoke_s = invoke.join(",");
267
268        let mut active: Vec<&str> = self.matched.iter().map(|m| m.as_str()).collect();
269        active.sort_unstable();
270        let active_s = active.join(",");
271
272        let mut data_pairs: Vec<_> = self
273            .filtered_datatypes
274            .iter()
275            .map(|(name, dt)| (name.as_str(), dt))
276            .collect();
277        data_pairs.sort_by_key(|(name, _)| *name);
278        let data_s = data_pairs
279            .into_iter()
280            .map(|(_, dt)| dt.to_string())
281            .collect::<Vec<_>>()
282            .join(",");
283
284        let actions_s = if self.actions.drop() {
285            String::new()
286        } else {
287            format!("{}", self.actions)
288        };
289
290        let mut parts: Vec<String> = Vec::new();
291        if !actions_s.is_empty() {
292            parts.push(format!("Actions: {}", actions_s));
293        }
294        if !invoke_s.is_empty() {
295            parts.push(format!("Invoke: {}", invoke_s));
296        }
297        if !active_s.is_empty() {
298            parts.push(format!("Active: {}", active_s));
299        }
300        if !data_s.is_empty() {
301            parts.push(format!("Data: {}", data_s));
302        }
303
304        let mut pred_s = format!("{}", self.pred);
305        if self.if_else {
306            pred_s.push_str(" x");
307        }
308
309        if parts.is_empty() {
310            return write!(f, "{}", pred_s);
311        }
312
313        let candidate_single = format!("{}  {}", pred_s, parts.join("  "));
314        if candidate_single.len() <= 100 {
315            return write!(f, "{}", candidate_single);
316        }
317
318        // Multi-line: predicate header, then section lines (indent handled by tree printer).
319        write!(f, "{}", pred_s)?;
320        for p in parts {
321            write!(f, "\n{}", p)?;
322        }
323        Ok(())
324    }

347// Used for ordering nodes by protocol and field
348// Does NOT consider contents of the node
349impl Ord for PNode {
350    fn cmp(&self, other: &PNode) -> Ordering {
351        if let Predicate::Binary {
352            protocol: proto,
353            field: field_name,
354            op: _op,
355            value: _val,
356        } = &self.pred
357        {
358            if let Predicate::Binary {
359                protocol: peer_proto,
360                field: peer_field_name,
361                op: _peer_op,
362                value: _peer_val,
363            } = &other.pred
364            {
365                // Same protocol; sort fields
366                if proto == peer_proto {
367                    return field_name.name().cmp(peer_field_name.name());
368                }
369            }
370        }
371
372        // If either has a protocol name, sort by that
373        // This also will put all ProtocolName::none predicates
374        // (i.e., LayerState or user-defined) next to each other
375        if self.pred.get_protocol() != ProtocolName::none()
376            || other.pred.get_protocol() != ProtocolName::none()
377        {
378            return self
379                .pred
380                .get_protocol()
381                .name()
382                .cmp(other.pred.get_protocol().name());
383        }
384
385        // Both predicates are custom filters, callbacks, or state checks;
386        // compare their string representations
387        format!("{}", self.pred).cmp(&format!("{}", other.pred))
388    }
389}

391// A n-ary tree representing a Filter.
392// Paths from root to leaf represent a pattern for data to match.
393// Filter returns action(s) or delivers data.
394#[derive(Debug, Clone)]
395pub struct PTree {
396    // Root node
397    pub root: PNode,
398
399    // Number of nodes in tree
400    pub size: usize,
401
402    // Which filter this PTree represents
403    pub filter_layer: StateTransition,
404
405    // Has `collapse` been applied?
406    // Use to ensure no filters are applied after `collapse`
407    collapsed: bool,
408
409    // All actions, callbacks, tracked datatypes across all nodes in the tree
410    pub actions: NodeActions,
411    pub deliver: HashSet<CallbackSpec>,
412    pub matched: HashSet<String>,
413    pub filtered_datatypes: HashSet<String>,
414}

416impl PTree {
417    pub fn new_empty(filter_layer: StateTransition) -> Self {

420            size: 1,

427        }

430    pub fn add_subscription(
431        &mut self,
432        // Filter patterns
433        patterns: &[FlatPattern],
434        // Individual callbacks; each has datatypes
435        callbacks: &[CallbackSpec],
436        // String identifier
437        id: &str,
438    ) {
439        if self.collapsed {
440            panic!("Cannot add filter to tree after collapsing");
441        }
442        assert!(
443            !patterns.is_empty() || patterns.iter().all(|p| p.predicates.is_empty()),
444            "Empty filter pattern must have default predicate."
445        );
446        log::trace!("{}: Adding subscription with id: {}", self.filter_layer, id);
447        if patterns.is_empty() {
448            self.add_pattern(
449                &FlatPattern {
450                    predicates: vec![Predicate::default_pred()],
451                    as_str: None,
452                },
453                callbacks,
454                id,
455            );
456            return;
457        }
458
459        let ext_patterns = self.split_custom(patterns);
460        for pattern in patterns.iter().chain(ext_patterns.iter()) {
461            self.add_pattern(pattern, callbacks, id);
462        }
463    }
464
465    // Add patterns for non-terminal matches on streaming filters
466    // NICE-TO-HAVE: may not always need this (if no correctness issues?)
467    fn split_custom(&self, patterns: &[FlatPattern]) -> Vec<FlatPattern> {
468        let mut split_patterns = vec![];
469        if matches!(self.filter_layer, StateTransition::L4Terminated) {
470            return split_patterns;
471        }
472        for pattern in patterns {
473            split_patterns.extend(pattern.split_custom());
474        }
475        split_patterns
476    }
477
478    fn add_pattern(&mut self, pattern_: &FlatPattern, callbacks: &[CallbackSpec], cb_id: &str) {
479        let mut pattern = pattern_;
480
481        // Pattern may have predicates that need to be annotated with
482        // L7 State checks (e.g., only check TLS if L7 >= Headers)
483        let mut pattern_tmp;
484        if self.filter_layer.is_streaming() && self.filter_layer.in_transport() {
485            pattern_tmp = pattern.with_l7_state();
486            pattern = &pattern_tmp;
487        }
488        // Extend with callback predicates if this callback will be stateful
489        // (multiple functions or streaming).
490        // If a subscription previously had the opportunity to unsubscribe,
491        // we need to check if it did
492        if !matches!(
493            self.filter_layer,
494            StateTransition::L4FirstPacket | StateTransition::L4Terminated
495        ) && (callbacks.len() > 1
496            || callbacks.iter().any(|c| match c.expl_level {
497                Some(l) => l.is_streaming(),
498                None => false,
499            }))
500        {
501            pattern_tmp = pattern.with_streaming_cb(cb_id);
502            pattern = &pattern_tmp;
503        }
504
505        // Extract required datatypes in filter pattern
506        let mut datatypes = pattern.get_datatypes();
507        // Add datatypes required by callbacks
508        for callback in callbacks {
509            datatypes.extend(callback.get_datatypes());
510        }
511        // Construct node actions
512        let mut node_actions = NodeActions::new(self.filter_layer);
513        // Actions required for all datatypes (including filters)
514        for spec in &datatypes {
515            node_actions.add_datatype(spec);
516        }
517        // Update `refresh_at` based on where next filter predicate(s)
518        // may be applied.
519        node_actions.end_datatypes();
520        for next_pred in pattern.next_pred(self.filter_layer) {
521            node_actions.push_filter_pred(&next_pred);
522        }
523
524        // Allow callback to `unsubscribe`
525        for callback in callbacks {
526            if let Some(cb_level) = &callback.expl_level {
527                node_actions.push_cb(*cb_level);
528            }
529        }
530
531        let contains_nonterminal = pattern.contains_nonterminal();
532        let mut full_pattern_added = false;
533
534        // Add dependencies separately
535        for action in &node_actions.actions {
536            let mut pattern_ref = pattern;
537            let subpattern: FlatPattern;
538            let mut truncated = contains_nonterminal;
539
540            // Need to insert this as a unique pattern with the relevant L7 state checks
541            if let Some(if_matches) = action.if_matches {
542                subpattern = pattern.get_subpattern(if_matches.0, if_matches.1);
543                // Subpattern should have added a predicate (L7 state check)
544                truncated = truncated || subpattern.predicates.len() < pattern.predicates.len() + 1;
545                pattern_ref = &subpattern;
546            }
547
548            // If all patterns get (retroactively) truncated, we'll need to
549            // make sure (at the end) that callbacks are delivered if possible.
550            full_pattern_added |= !truncated;
551
552            // Add pattern for each callback
553            for callback in callbacks {
554                self.add_pattern_int(pattern_ref, action, callback, truncated);
555            }
556        }
557
558        if !contains_nonterminal && (!full_pattern_added || node_actions.actions.is_empty()) {
559            for callback in callbacks {
560                let level =
561                    SubscriptionLevel::new(&callback.datatypes, pattern, callback.expl_level);
562                if level.can_deliver(&self.filter_layer) {
563                    self.add_pattern_int(
564                        pattern,
565                        &DataActions::new(),
566                        callback,
567                        contains_nonterminal,
568                    );
569                }
570            }
571        }
572    }
573
574    fn add_pattern_int(
575        &mut self,
576        full_pattern: &FlatPattern,
577        actions: &DataActions,
578        callback: &CallbackSpec,
579        // `full_pattern` was truncated
580        mut truncated: bool,
581    ) {
582        let level = SubscriptionLevel::new(&callback.datatypes, full_pattern, callback.expl_level);
583        if level.can_skip(&self.filter_layer) {
584            return;
585        }
586        let mut state_pred = None;
587
588        let pattern = FlatPattern {
589            predicates: full_pattern
590                .predicates
591                .iter()
592                .filter(|p| !p.is_next_layer(self.filter_layer))
593                .cloned()
594                .collect(),
595            as_str: full_pattern.as_str.clone(),
596        };
597        assert!(pattern.predicates.len() <= full_pattern.predicates.len());
598        if pattern.predicates.len() < full_pattern.predicates.len() {
599            assert!(
600                !level.can_deliver(&self.filter_layer),
601                "Ensure that {} explicit level {:?} is compatible with filter string {}",
602                callback.as_str,
603                callback.expl_level,
604                full_pattern
605            );
606            truncated = true;
607        }
608
609        log::trace!(
610            "{}: Adding pattern {} with actions {} and callback {}",
611            self.filter_layer,
612            pattern,
613            actions,
614            callback.as_str
615        );
616
617        let mut node = &mut self.root;
618        for predicate in pattern.predicates.iter() {
619            // Case 1: Predicate is already present
620            if node.has_descendant(predicate, &state_pred) {
621                node = node.get_descendant(predicate).unwrap();
622                continue;
623            }
624            // Case 2: Predicate should be added as child of existing node
625            if node.has_parent(predicate) {
626                node = node.get_parent(predicate, self.size, &state_pred).unwrap();
627            }
628            // Case 3: Children of curr node should be children of new node
629            let children = match node.has_children_of(predicate) {
630                true => node.get_children_of(predicate),
631                false => {
632                    vec![]
633                }
634            };
635            // Create new node
636            if !node.has_child(predicate) {
637                node.children.push(PNode::new(predicate.clone(), self.size));
638                self.size += 1;
639            }
640            // Move on, pushing any new children if applicable
641            node = node.get_child(predicate);
642            node.children.extend(children);
643
644            // Maintain streaming filter if still `matching` regardless of
645            // later predicate state
646            // NICE-TO-HAVE: do we need this? could avoid additional lookups
647            if node.pred.is_custom() && node.pred.is_matching() {
648                node.actions.merge(&DataActions::from_stream_pred(
649                    &node.pred,
650                    full_pattern.next_pred(self.filter_layer),
651                    self.filter_layer,
652                    &state_pred,
653                ));
654            }
655
656            if node.pred.is_state() {
657                // Used to determine when we can/can't "skip ahead"
658                // NICE-TO-HAVE: may not need this - may just be able to stop traversing in opts if
659                // any state pred is hit
660                state_pred = Some(node.pred.clone());
661            }
662        }
663        if !truncated && level.can_deliver(&self.filter_layer) {
664            node.deliver.insert(callback.clone());
665            self.deliver.insert(callback.clone());
666        } else if !truncated
667            && (callback.is_streaming() || callback.is_grouped())
668            && pattern.is_first_match(&self.filter_layer)
669        {
670            // Can't deliver, but this might be the first time this pattern
671            // has matched, so we need to mark the callback as active
672            node.matched.insert(callback.subscription_id.clone());
673            self.matched.insert(callback.subscription_id.clone());
674        }
675
676        if !truncated && pattern.as_str.is_some() {
677            assert!(node.filter_str.is_none() || node.filter_str == pattern.as_str);
678            node.filter_str = pattern.as_str.clone();
679        }
680
681        node.actions.merge(actions);
682        let filtered_data_names: Vec<String> = pattern
683            .get_filtered_data()
684            .iter()
685            .chain(callback.filtered_data.iter())
686            .cloned()
687            .collect();
688
689        if !filtered_data_names.is_empty() {
690            self.filtered_datatypes
691                .extend(filtered_data_names.iter().cloned());
692            for dt in filtered_data_names {
693                let data = FilteredDatatype::new(dt.clone(), actions.refresh_at());
694                node.insert_filtered_data(data);
695            }
696        }
697

701    // modified from https://vallentin.dev/2019/05/14/pretty-print-tree
702    fn pprint(&self) -> String {
703        fn pprint(s: &mut String, node: &PNode, prefix: String, last: bool) {
704            let prefix_current = if last { "`- " } else { "|- " };
705            let prefix_child = if last { "   " } else { "|  " };
706
707            let node_str = format!("{}", node);
708            let mut lines = node_str.lines();
709            let first = lines.next().unwrap_or("");
710            s.push_str(&format!(
711                "{}{}{}: {}\n",
712                prefix, prefix_current, node.id, first
713            ));
714
715            // Line breaks in node's printout need to be indented
716            // Follow the tree indentation, plus 2 extra spaces for data
717            let cont_prefix = format!(
718                "{}{}{}",
719                prefix,
720                prefix_child,
721                " ".repeat(node.id.to_string().len() + 2) // ": "
722            );
723            for line in lines {
724                s.push_str(&format!("{}  {}\n", cont_prefix, line));
725            }
726
727            let prefix = prefix + prefix_child;
728
729            if !node.children.is_empty() {
730                // "last child" in a tree layer formatted with `-
731                let last_child = node.children.len() - 1;
732
733                for (i, child) in node.children.iter().enumerate() {
734                    pprint(s, child, prefix.to_string(), i == last_child);
735                }
736            }
737        }
738
739        let mut s = String::new();
740        pprint(&mut s, &self.root, "".to_string(), true);
741        s
742    }

744    // Returns a copy of the subtree rooted at Node `id`
745    pub fn get_subtree(&self, id: usize) -> Option<PNode> {
746        fn get_subtree(id: usize, node: &PNode) -> Option<PNode> {
747            if node.id == id {
748                return Some(node.clone());
749            }
750            for child in node.children.iter() {
751                if let Some(node) = get_subtree(id, child) {
752                    return Some(node);
753                }
754            }
755            None
756        }
757        get_subtree(id, &self.root)
758    }

775    // Sorts the PTree according to predicates
776    // Useful as a pre-step for marking mutual exclusion; places
777    // conditions with the same protocols/fields next to each other.
778    fn sort(&mut self) {
779        fn sort(node: &mut PNode) {
780            for child in node.children.iter_mut() {
781                sort(child);
782            }
783            node.children.sort();
784        }
785        sort(&mut self.root);

788    // Best-effort to give the filter generator hints as to where an "else"
789    // statement can go between two predicates. That is, if branches A and B
790    // are mutually exclusive, we want the runtime to take at most one of them.
791    fn mark_mutual_exclusion(&mut self) {
792        fn mark_mutual_exclusion(node: &mut PNode) {
793            for idx in 0..node.children.len() {
794                // Recurse for children/descendants
795                mark_mutual_exclusion(&mut node.children[idx]);
796                if idx == 0 {
797                    continue;
798                }
799                // Look for mutually exclusive predicates in direct children
800                if node.children[idx]
801                    .pred
802                    .is_excl(&node.children[idx - 1].pred)
803                {
804                    node.children[idx].if_else = true;
805                }
806                // If the result is equivalent (e.g., same actions)
807                // for child nodes, then we can safely use first match.
808                // (Similar to "early return.")
809                if node.children[idx].outcome_eq(&node.children[idx - 1]) {
810                    node.children[idx].if_else = true;
811                }
812            }
813        }
814        mark_mutual_exclusion(&mut self.root);
815    }

817    // Removes some patterns that are covered by others
818    fn prune_branches(&mut self) {
819        fn prune(
820            node: &mut PNode,
821            on_path_actions: &DataActions,
822            on_path_deliver: &HashSet<String>,
823            on_path_matched: &HashSet<String>,
824            on_path_data: &HashMap<String, FilteredDatatype>,
825        ) {
826            // 1. Remove callbacks that would have already been invoked on this path
827            let mut my_deliver = on_path_deliver.clone();
828            let mut new_ids = HashSet::new();
829            for i in &node.deliver {
830                if !my_deliver.contains(&i.as_str) {
831                    my_deliver.insert(i.as_str.clone());
832                    new_ids.insert(i.clone());
833                } else if i.must_deliver {
834                    new_ids.insert(i.clone());
835                }
836            }
837            node.deliver = new_ids;
838
839            // 2. Do the same for actions
840            let mut my_actions = on_path_actions.clone();
841            if !node.actions.drop() {
842                node.actions.clear_intersection(&my_actions);
843                my_actions.merge(&node.actions);
844            }
845
846            // 3. Do the same for callbacks that need to be marked `matched`
847            let mut my_matched = on_path_matched.clone();
848            let mut new_ids = HashSet::new();
849            for i in &node.matched {
850                if !my_matched.contains(i) {
851                    my_matched.insert(i.clone());
852                    new_ids.insert(i.clone());
853                }
854            }
855            node.matched = new_ids;
856
857            // 4. ...and for filtered data
858            let mut my_filtered_data = on_path_data.clone();
859            let mut new_data = HashMap::new();
860            for dt in node.filtered_datatypes.values() {
861                match my_filtered_data.get(&dt.name) {
862                    Some(on_path_dt) => {
863                        // Datatype already present on path
864                        if !dt.is_contained_in(on_path_dt) {
865                            // ...but has different `refresh_at` fields
866                            my_filtered_data.insert(dt.name.clone(), dt.clone());
867                            new_data.insert(dt.name.clone(), dt.clone());
868                        }
869                    }
870                    None => {
871                        my_filtered_data.insert(dt.name.clone(), dt.clone());
872                        new_data.insert(dt.name.clone(), dt.clone());
873                    }
874                }
875            }
876            node.filtered_datatypes = new_data;
877
878            // 5. Repeat for each child
879            node.children.iter_mut().for_each(|child| {
880                prune(
881                    child,
882                    &my_actions,
883                    &my_deliver,
884                    &my_matched,
885                    &my_filtered_data,
886                )
887            });
888
889            // 6. Remove empty children
890            let children = std::mem::take(&mut node.children);
891            node.children = children
892                .into_iter()
893                .filter(|child| child.has_op() || !child.children.is_empty())
894                .collect();
895        }
896
897        let on_path_actions = DataActions::new();
898        let on_path_deliver = HashSet::new();
899        let on_path_matched = HashSet::new();
900        let on_path_data = HashMap::new();
901        prune(
902            &mut self.root,
903            &on_path_actions,
904            &on_path_deliver,
905            &on_path_matched,
906            &on_path_data,
907        );
908    }

910    // Avoid re-checking packet-level conditions that, on the basis of previous
911    // filters, are guaranteed to be already met.
912    // For example, if all subscriptions filter for "tcp", then all non-tcp
913    // connections will have been filtered out at the PacketContinue layer.
914    // We only do this for packet-level conditions, as connection-level
915    // conditions are needed to extract sessions.
916    fn prune_packet_conditions(&mut self) {
917        fn prune_packet_conditions(
918            node: &mut PNode,
919            filter_layer: StateTransition,
920            can_prune: bool,
921        ) {
922            if !node.pred.on_packet() {
923                return;
924            }
925            // Can only safely remove children if
926            // current branches are mutually exclusive
927            let can_prune_next = node
928                .children
929                .windows(2)
930                .all(|w| w[0].pred.is_excl(&w[1].pred));
931            for child in &mut node.children {
932                prune_packet_conditions(child, filter_layer, can_prune_next);
933            }
934            if !can_prune {
935                return;
936            }
937            // Tree layer is only drop/keep (i.e., one condition),
938            // and condition checked at prev. layer
939            while node.children.len() == 1 && node.children[0].pred.on_packet() {
940                // If the protocol needs to be extracted, can't remove node
941                // Look for unary predicate (e.g., `ipv4`) and child with
942                // binary predicate of same protocol (e.g., `ipv4.addr = ...`)
943                let child = &mut node.children[0];
944                if child.extracts_protocol(filter_layer)
945                    || child.pred.is_state()
946                    || child.pred.is_callback()
947                {
948                    break;
949                }
950                node.actions.merge(&child.actions);
951                node.deliver.extend(child.deliver.iter().cloned());
952                node.filtered_datatypes =
953                    FilteredDatatype::concat(&node.filtered_datatypes, &child.filtered_datatypes);
954                node.matched.extend(child.matched.iter().cloned());
955                node.children = std::mem::take(&mut child.children);
956            }
957        }
958
959        let can_prune_next = self
960            .root
961            .children
962            .windows(2)
963            .all(|w| w[0].pred.is_excl(&w[1].pred));
964        prune_packet_conditions(&mut self.root, self.filter_layer, can_prune_next);
965    }

967    // Avoid applying conditions that (1) are not needed for filtering *out*
968    // (i.e., would have already been checked by prev layer), and (2) end in
969    // the same result.
970    // Example: two different IP addresses in a packet filter followed by
971    // a TCP/UDP disambiguation.
972    fn prune_redundant_branches(&mut self) {
973        fn prune_redundant_branches(
974            node: &mut PNode,
975            filter_layer: StateTransition,
976            can_prune: bool,
977        ) {
978            if !node.pred.is_prev_layer(filter_layer) {
979                return;
980            }
981
982            // Can only safely remove children if
983            // current branches are mutually exclusive
984            let can_prune_next = node
985                .children
986                .windows(2)
987                .all(|w| w[0].pred.is_excl(&w[1].pred));
988
989            for child in &mut node.children {
990                prune_redundant_branches(child, filter_layer, can_prune_next);
991            }
992            if !can_prune {
993                return;
994            }
995
996            let (must_keep, could_drop): (Vec<PNode>, Vec<PNode>) =
997                node.children.iter().cloned().partition(|child| {
998                    child.has_op()
999                        || !child.pred.is_prev_layer(filter_layer)
1000                        || child.extracts_protocol(filter_layer)
1001                        || child.pred.is_state()
1002                });
1003            let mut new_children = vec![];
1004            for child in &could_drop {
1005                // Can "upgrade" descendants if all children in a layer
1006                // have the same descendant conditions.
1007                if node.children.iter().all(|c| child.all_paths_eq(c)) {
1008                    new_children.extend(child.children.clone());
1009                } else {
1010                    new_children.push(child.clone());
1011                }
1012            }
1013
1014            new_children.extend(must_keep);
1015            new_children.sort();
1016            new_children.dedup();
1017            node.children = new_children;
1018        }
1019        let can_prune_next = self
1020            .root
1021            .children
1022            .windows(2)
1023            .all(|w| w[0].pred.is_excl(&w[1].pred));
1024        prune_redundant_branches(&mut self.root, self.filter_layer, can_prune_next);
1025    }

1027    // After collapsing the tree, make sure node IDs and sizes are correct.
1028    fn update_size(&mut self) {
1029        fn count_nodes(node: &mut PNode, id: &mut usize) -> usize {
1030            node.id = *id;
1031            *id += 1;
1032            let mut count = 1;
1033            for child in &mut node.children {
1034                count += count_nodes(child, id);
1035            }
1036            count
1037        }
1038        let mut id = 0;
1039        self.size = count_nodes(&mut self.root, &mut id);
1040    }

1042    // Update tree actions, filtered datatypes, callbacks, and size based on children
1043    fn update_metadata(&mut self) {
1044        fn update_metadata(
1045            node: &PNode,
1046            actions: &mut NodeActions,
1047            deliver: &mut HashSet<CallbackSpec>,
1048            matched: &mut HashSet<String>,
1049            filtered_datatypes: &mut HashSet<String>,
1050        ) {
1051            for child in &node.children {
1052                update_metadata(child, actions, deliver, matched, filtered_datatypes);
1053            }
1054            actions.push_action(node.actions.clone());
1055            deliver.extend(node.deliver.iter().cloned());
1056            matched.extend(node.matched.iter().cloned());
1057            filtered_datatypes.extend(node.filtered_datatypes.keys().cloned());
1058        }
1059        self.update_size();
1060        self.actions = NodeActions::new(self.filter_layer);
1061        self.deliver.clear();
1062        self.matched.clear();
1063        self.filtered_datatypes.clear();
1064        update_metadata(
1065            &self.root,
1066            &mut self.actions,
1067            &mut self.deliver,
1068            &mut self.matched,
1069            &mut self.filtered_datatypes,
1070        );
1071    }

1073    // Apply all filter tree optimizations.
1074    // This must only be invoked AFTER the tree is completely built.
1075    pub fn collapse(&mut self) {
1076        if matches!(self.filter_layer, StateTransition::L4Terminated) {
1077            self.collapsed = true;
1078            // Shouldn't have another filter stage here unless there may
1079            // be something to deliver
1080            assert!(!self.deliver.is_empty());
1081
1082            // The delivery filter will only be invoked if a previous filter
1083            // determined that delivery is needed at the corresponding stage.
1084            // If disambiguation is not needed (i.e., only one possible delivery
1085            // outcome), then no filter condition is needed.
1086            // The exception is if there is a filter that we still need to apply.
1087            // --> NICE-TO-HAVE: can we relax this requirement?
1088            if self.deliver.len() == 1 && !self.contains_term_filters() {
1089                self.root
1090                    .deliver
1091                    .insert(self.deliver.iter().next().unwrap().clone());
1092                self.root.children.clear();
1093                self.update_size();
1094                return;
1095            }
1096        }
1097        self.prune_redundant_branches();
1098        self.prune_packet_conditions();
1099        self.prune_branches();
1100        self.sort();
1101        self.mark_mutual_exclusion(); // Must be last
1102        self.update_metadata();
1103        // TODO FOR FILTER_STR: will "collapse" ever mess up the "FilterStr"?
1104    }

1106    /// REFACTOR: this is messy
1107    ///
1108    /// Insert a node that forces the code generator to extract the last matched session (if one exists)
1109    /// so that a child node can match on it.
1110    /// This is a quirk of the generated code; see `extract_sessions` in `compiler/src/state_filter.rs`.
1111    pub fn extract_sessions(&mut self) {
1112        // Indicates that there is some child that requires session extraction;
1113        // candidate for inserting session extractor node
1114        fn session_state(node: &PNode) -> bool {
1115            matches!(
1116                node.pred,
1117                Predicate::LayerState {
1118                    layer: SupportedLayer::L7,
1119                    state: LayerState::Headers,
1120                    op: super::ast::BinOp::Gt,
1121                } | Predicate::LayerState {
1122                    layer: SupportedLayer::L7,
1123                    state: LayerState::Payload,
1124                    op: super::ast::BinOp::Eq | super::ast::BinOp::Ge,
1125                }
1126            )
1127        }
1128
1129        // Confirm that a child (or descendant) is actually on_session
1130        fn has_on_session(node: &PNode) -> bool {
1131            node.pred.on_session() || node.children.iter().any(has_on_session)
1132        }
1133
1134        // Confirm that a child (or descendent) has an on_proto
1135        // before any on_session
1136        fn has_on_proto(node: &PNode) -> bool {
1137            if node.pred.on_session() {
1138                return false;
1139            } // early return
1140            node.pred.on_proto() || node.children.iter().any(has_on_proto)
1141        }
1142
1143        fn insert_session_node(node: &mut PNode, proto: Option<ProtocolName>) {
1144            // Record protocol
1145            let proto = if node.pred.on_proto() {
1146                Some(node.pred.get_protocol().clone())
1147            } else {
1148                proto
1149            };
1150
1151            if node.pred.is_state() && session_state(node) && has_on_session(node) {
1152                if proto.is_none() {
1153                    assert!(has_on_proto(node), "No protocol found on path to {}", node);
1154                    return;
1155                }
1156                let mut proto_node = PNode::new(
1157                    Predicate::Unary {
1158                        protocol: proto.as_ref().unwrap().clone(),
1159                    },
1160                    0, // updated in `update_size`
1161                );
1162                proto_node.children = std::mem::take(&mut node.children);
1163                node.children = vec![proto_node];
1164                // Early return; nothing else to do for children
1165                return;
1166            }
1167
1168            for child in &mut node.children {
1169                insert_session_node(child, proto.clone());
1170            }
1171        }
1172
1173        // Might have some LayerState predicates that require extracting the session
1174        // (e.g., TLS handshake) before we can check fields on it.
1175        if self.filter_layer.is_streaming() && self.filter_layer.in_transport() {
1176            let proto = None;
1177            insert_session_node(&mut self.root, proto);
1178            self.update_size();
1179        }
1180    }
