5/// Interface for datatypes that must be "tracked" throughout
6/// all or part of a connection.
7/// Datatypes tagged with "track" must implement this trait.
8///
9/// Each tracked datatype can optionally be tagged as "track", which
10/// indicates that the runtime should track which subscriptions require
11/// it and drop the Tracked data if all of those subscriptions go out
12/// of scope. This may limit how much the compiler can optimize the
13/// filter predicates, but it is generally valuable if the datatype is memory-
14/// or computationally-intensive (e.g., a list of packets) and is only needed
15/// for a small subset of connections.
16pub trait Tracked {
17    /// Initialize internal data. Invoked on first PDU in connection.
18    /// Note that this first PDU will also be received in `update`.
19    fn new(first_pkt: &L4Pdu) -> Self;
20    /// Utility method to clear internal data.
21    /// Recommended to implement for memory-intensive datatypes.
22    fn clear(&mut self);
23}
24
25/// Can be requested in a callback. Represents the string literal
26/// of the filter pattern (i.e., list of conjunctive predicates)
27/// that triggered invocation of the callback.
28/// This may be useful if one callback specifies multiple filter patterns.
29///
30/// Examples:
31/// - Filter: "tls or quic" parsed as patterns
32///   `["eth->tcp->tls", "eth->udp->quic"]`
33/// - "tls.sni contains nflx or tls.sni contains hulu" parsed as patterns
34///   `["eth->tcp->tls.sni contains nflx", "eth->tcp->tls.sni contains hulu"]`
35///
36/// Some "gotchas":
37/// - FilterStr cannot be requested in a streaming callback (InL4Conn).
38///   If you need FilterStr along with streaming data, we recommend defining
39///   a stateful callback with two methods.
40/// - Filter patterns may not be mutually exclusive; i.e., multiple patterns
41///   can possibly match. Iris guarantees that stateless, non-streaming
42///   callbacks are invoked at most once; if multiple patterns match, the FilterStr
43///   delivered will be the one that matches first. Iris makes no such guarantee
44///   for stateful callbacks. If you want to receive every pattern that matches,
45///   if multiple match, define a stateful callback.
46pub type FilterStr<'a> = &'a str;

48/// Must be implemented as a trait; cannot define inherent `impl`
49/// for foreign type.
50#[doc(hidden)]
51pub trait StringToTokens {
52    fn from_string(filter: &str) -> proc_macro2::TokenStream;
53}
54
55impl StringToTokens for FilterStr<'_> {
56    /// Convert a filter string into a token representation at compile-time.
57    #[doc(hidden)]
58    fn from_string(filter: &str) -> proc_macro2::TokenStream {
59        let str = syn::LitStr::new(filter, proc_macro2::Span::call_site());
60        quote! { &#str }
61    }
62}

64type RefreshAtState = BitArray<[u8; 1], Lsb0>;
65
66#[derive(Debug)]
67#[doc(hidden)]
68pub enum DatatypeState {
69    /// Should be updated
70    Active,
71    /// Pending (in state transition)
72    Pending,
73    /// No longer active at all
74    Inactive,
75}
76
77/// If a datatype is specified as "expensive", it is wrapped in this
78/// so that we can track when it's still needed by active subscriptions.
79#[doc(hidden)]
80pub struct TrackedDataWrapper<T>
81where
82    T: Tracked + std::fmt::Debug,
83{
84    /// The wrapped tracked data.
85    pub data: T,
86    /// Whether and until when to continue updating.
87    state: DatatypeState,
88    /// States to "refresh at"
89    refresh_at: RefreshAtState,
90}
91
92impl<T> TrackedDataWrapper<T>
93where
94    T: Tracked + std::fmt::Debug,
95{
96    pub fn new(pdu: &L4Pdu) -> Self {
97        Self {
98            data: T::new(pdu),
99            state: DatatypeState::Active,
100            refresh_at: BitArray::ZERO,
101        }
102    }
103
104    pub fn clear(&mut self) {
105        self.data.clear();
106        self.state = DatatypeState::Inactive;
107        self.refresh_at.fill(false);
108    }
109
110    pub fn start_state_tx(&mut self, tx: &StateTransition) {
111        if matches!(self.state, DatatypeState::Active) && self.refresh_at[tx.as_usize()] {
112            self.state = DatatypeState::Pending;
113        }
114    }
115
116    pub fn set_active_until(&mut self, txs: u8) {
117        self.state = DatatypeState::Active;
118        self.refresh_at.as_raw_mut_slice()[0] |= txs;
119    }
120
121    pub fn is_active(&self) -> bool {
122        // Include `pending` so that functions can still be invoked within State TX
123        // handlers, even as data may be filtered out
124        matches!(self.state, DatatypeState::Active | DatatypeState::Pending)
125    }
126
127    pub fn end_state_tx(&mut self) {
128        // Was not set to be "active"
129        if matches!(self.state, DatatypeState::Pending) {
130            self.clear();
131        }
132    }
133}
