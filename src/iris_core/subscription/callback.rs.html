3/// The framework expects that any stateful callback implements this trait.
4/// The user must also define the actual callback function(s), annotated with
5/// the appropriate #[callback_fn(...)] macros.
6pub trait StreamingCallback {
7    /// Initializes internal data, if applicable.
8    /// Called on first packet in connection *whether or not callback filter has matched*.
9    fn new(first_pkt: &L4Pdu) -> Self;
10    /// Clears internal data, if applicable. Called when callback unsubscribes.
11    fn clear(&mut self);
12}
13
14#[doc(hidden)]
15#[derive(Debug)]
16pub enum CallbackState {
17    Active,
18    Matching,
19    Unsubscribed,
20}
21
22#[doc(hidden)]
23/// Wrapper for a stateful callback that is invoked in a streaming state
24/// (e.g., InL4Conn) or has multiple functions.
25#[derive(Debug)]
26pub struct StreamCallbackWrapper<C>
27where
28    C: StreamingCallback + std::fmt::Debug,
29{
30    state: CallbackState,
31    pub callback: C,
32}
33
34impl<C> StreamCallbackWrapper<C>
35where
36    C: StreamingCallback + std::fmt::Debug,
37{
38    pub fn new(first_pkt: &L4Pdu) -> Self {
39        Self {
40            state: CallbackState::Matching,
41            callback: C::new(first_pkt),
42        }
43    }
44
45    /// Returns true if the callback should be invoked.
46    pub fn is_active(&self) -> bool {
47        matches!(self.state, CallbackState::Active)
48    }
49
50    /// Invoked when a filter pattern for callback has matched.
51    pub fn try_set_active(&mut self) {
52        if !matches!(self.state, CallbackState::Unsubscribed) {
53            self.state = CallbackState::Active;
54        }
55    }
56
57    /// Invoked when the callback has unsubscribed.
58    pub fn set_inactive(&mut self) {
59        if matches!(self.state, CallbackState::Active) {
60            self.callback.clear();
61        }
62        self.state = CallbackState::Unsubscribed;
63    }
64
65    /// Invoked when a callback goes out-of-scope
66    pub fn clear(&mut self) {
67        self.callback.clear();
68    }
69}

92            self.state = CallbackState::Active;

102/// Wrapper around a boolean value to help ensure that a
103/// callback is invoked once per connection or session if that
104/// is its expected behavior. The framework uses this wrapper when
105/// a subscription specifies a streaming filter and a non-streaming
106/// callback.
107#[derive(Debug)]
108pub struct StaticCallbackWrapper {
109    pub invoked: bool,
110}
111impl StaticCallbackWrapper {
112    // Note - easier for code generation if all `new` fns
113    // take in `L4Pdu` ref.
114    pub fn new(_: &L4Pdu) -> Self {
115        Self { invoked: false }
116    }
117    pub fn should_invoke(&self) -> bool {
118        !self.invoked
119    }
120    pub fn set_invoked(&mut self) {
121        self.invoked = true;
122    }
