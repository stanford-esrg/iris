20/// Pure SYN
21pub(crate) const HIST_SYN: u8 = b'S';
22/// Pure SYNACK
23pub(crate) const HIST_SYNACK: u8 = b'H';
24/// Pure ACK (no payload)
25pub(crate) const HIST_ACK: u8 = b'A';
26/// Has non-zero payload length
27pub(crate) const HIST_DATA: u8 = b'D';
28/// Has FIN set
29pub(crate) const HIST_FIN: u8 = b'F';
30/// Has RST set
31pub(crate) const HIST_RST: u8 = b'R';

33impl ConnRecord {
34    /// Returns the client (originator) socket address.
35    #[inline]
36    pub fn client(&self) -> SocketAddr {
37        self.five_tuple.orig
38    }
39
40    /// Returns the server (responder) socket address.
41    #[inline]
42    pub fn server(&self) -> SocketAddr {
43        self.five_tuple.resp
44    }
45
46    /// Returns the total number of packets observed in the connection.
47    #[inline]
48    pub fn total_pkts(&self) -> u64 {
49        self.orig.nb_pkts + self.resp.nb_pkts
50    }
51
52    /// Returns the total number of packet bytes observed, including headers and malformed packets.
53    #[inline]
54    pub fn total_pkt_bytes(&self) -> u64 {
55        self.orig.nb_pkt_bytes + self.resp.nb_pkt_bytes
56    }
57
58    /// Returns the total number of payload bytes observed, excluding those from malformed packets.
59    #[inline]
60    pub fn total_payload_bytes(&self) -> u64 {
61        self.orig.nb_payload_bytes + self.resp.nb_payload_bytes
62    }
63
64    /// Returns the connection history.
65    #[inline]
66    pub fn history(&self) -> String {
67        String::from_utf8_lossy(&self.history).into_owned()
68    }
69
70    /// Returns te duration of the connection.
71    ///
72    /// ## Remarks
73    /// This does not represent the actual duration of the connection in offline analysis. It
74    /// approximates the elapsed time between observation of the first and last observed packet in
75    /// the connection.
76    #[inline]
77    pub fn duration(&self) -> Duration {
78        if self.orig.nb_pkts + self.resp.nb_pkts == 1 {
79            return Duration::default();
80        }
81        self.last_seen_ts - self.first_seen_ts
82    }
83
84    /// The duration (approximate) between the first and second packets.
85    #[inline]
86    pub fn time_to_second_packet(&self) -> Duration {
87        if self.orig.nb_pkts + self.resp.nb_pkts == 1 {
88            return Duration::default();
89        }
90        self.second_seen_ts - self.first_seen_ts
91    }
92}

94impl Serialize for ConnRecord {
95    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
96    where
97        S: Serializer,
98    {
99        let mut state = serializer.serialize_struct("ConnRecord", 6)?;
100        state.serialize_field("five_tuple", &self.five_tuple)?;
101        state.serialize_field("duration", &self.duration())?;
102        state.serialize_field("time_to_second_pkt", &self.time_to_second_packet())?;
103        state.serialize_field("max_inactivity", &self.max_inactivity)?;
104        state.serialize_field("history", &self.history())?;
105        state.serialize_field("orig", &self.orig)?;
106        state.serialize_field("resp", &self.resp)?;
107        state.end()
108    }
109}

118/// Tracks a connection record throughout its lifetime.
119///
120/// ## Note
121/// Internal connection state is an associated type of a `pub` trait, and therefore must also be
122/// public. Documentation is hidden by default to avoid confusing users.
123#[derive(Debug, Clone)]
124#[cfg_attr(not(feature = "skip_expand"), datatype)]
125pub struct ConnRecord {
126    /// The connection 5-tuple.
127    pub five_tuple: FiveTuple,
128    /// Timestamp of the first packet.
129    ///
130    /// ## Remarks
131    /// This represents the time Iris observed the first packet in the connection, and does not
132    /// reflect timestamps read from a packet capture in offline analysis.
133    pub first_seen_ts: Instant,
134    /// Timestamp of the second packet (approximate).
135    pub second_seen_ts: Instant,
136    /// Timestamp of the last packet (approximate).
137    pub last_seen_ts: Instant,
138    /// Maximum duration of inactivity (the maximum time between observed segments).
139    pub max_inactivity: Duration,
140    // Connection history.
141    ///
142    /// This represents a summary of the connection history in the order the packets were observed,
143    /// with letters encoded as a vector of bytes. This is a simplified version of [state history in
144    /// Zeek](https://docs.zeek.org/en/v5.0.0/scripts/base/protocols/conn/main.zeek.html), and the
145    /// meanings of each letter are similar: If the event comes from the originator, the letter is
146    /// uppercase; if the event comes from the responder, the letter is lowercase.
147    /// - S: a pure SYN with only the SYN bit set (may have payload)
148    /// - H: a pure SYNACK with only the SYN and ACK bits set (may have payload)
149    /// - A: a pure ACK with only the ACK bit set and no payload
150    /// - D: segment contains non-zero payload length
151    /// - F: the segment has the FIN bit set (may have other flags and/or payload)
152    /// - R: segment has the RST bit set (may have other flags and/or payload)
153    ///
154    /// Each letter is recorded a maximum of once in either direction.
155    pub history: Vec<u8>,
156    /// Originator flow.
157    pub orig: Flow,
158    /// Responder flow.
159    pub resp: Flow,
160}
161
162#[inline]
163pub(crate) fn update_history(history: &mut Vec<u8>, segment: &L4Pdu, mask: u8) {
164    fn insert(history: &mut Vec<u8>, event: u8) {
165        if !history.contains(&event) {
166            history.push(event);
167        }
168    }
169    if segment.flags() == SYN {
170        insert(history, HIST_SYN ^ mask);
171    } else if segment.flags() == (SYN | ACK) {
172        insert(history, HIST_SYNACK ^ mask);
173    } else if segment.flags() == ACK && segment.length() == 0 {
174        insert(history, HIST_ACK ^ mask);
175    }
176
177    if segment.flags() & FIN != 0 {
178        insert(history, HIST_FIN ^ mask);
179    }
180    if segment.flags() & RST != 0 {
181        insert(history, HIST_RST ^ mask);
182    }
183    if segment.length() > 0 {
184        insert(history, HIST_DATA ^ mask);
185    }
186}

188impl ConnRecord {
189    #[inline]
190    fn update_data(&mut self, segment: &L4Pdu) {
191        let now = Instant::now();
192        let inactivity = now - self.last_seen_ts;
193        if inactivity > self.max_inactivity {
194            self.max_inactivity = inactivity;
195        }
196        self.last_seen_ts = now;
197
198        if segment.dir {
199            update_history(&mut self.history, segment, 0x0);
200            self.orig.insert_segment(segment);
201        } else {
202            update_history(&mut self.history, segment, 0x20);
203            self.resp.insert_segment(segment);
204        }
205
206        if self.orig.nb_pkts + self.resp.nb_pkts == 2 {
207            self.second_seen_ts = now;
208        }
209    }
210
211    #[cfg_attr(not(feature = "skip_expand"), datatype_fn("ConnRecord,level=InL4Conn"))]
212    pub fn update(&mut self, pdu: &L4Pdu) {
213        self.update_data(pdu);
214    }
215}

217impl Tracked for ConnRecord {
218    fn new(first_pkt: &L4Pdu) -> Self {
219        let five_tuple = FiveTuple::from_ctxt(&first_pkt.ctxt);
220        let now = Instant::now();
221        Self {
222            five_tuple,
223            first_seen_ts: now,
224            second_seen_ts: now,
225            last_seen_ts: now,
226            max_inactivity: Duration::default(),
227            history: Vec::with_capacity(16),
228            orig: Flow::new(),
229            resp: Flow::new(),
230        }
231    }
232
233    // Clear the more memory-intensive data structures
234    fn clear(&mut self) {
235        self.orig.chunks = Vec::with_capacity(0);
236        self.orig.gaps = HashMap::with_capacity(0);
237        self.resp.chunks = Vec::with_capacity(0);
238        self.resp.gaps = HashMap::with_capacity(0);
239        self.history = Vec::with_capacity(0);
240    }
241}

243/// Default value for maximum chunk capacity.
244const DEFAULT_CHUNK_CAPACITY: usize = 100;

246/// A uni-directional flow.
247#[derive(Debug, Clone, Serialize)]
248pub struct Flow {
249    /// Number of packets seen for this flow, including malformed and late start segments.
250    ///
251    /// - Malformed segments are defined as those that have a payload offset (start of the payload,
252    ///   as computed from the header length field) beyond the end of the packet buffer, or the end
253    ///   of the payload exceeds the end of the packet buffer.
254    /// - Late start segments are those that arrive after the first packet seen in the flow, but
255    ///   have an earlier sequence number. Only applies to TCP flows.
256    pub nb_pkts: u64,
257    /// Number of malformed packets.
258    pub nb_malformed_pkts: u64,
259    /// Number of late start packets.
260    pub nb_late_start_pkts: u64,
261    /// Number of packet bytes observed in the flow. Includes bytes from malformed
262    /// segments and ethernet and ip headers.
263    pub nb_pkt_bytes: u64,
264    /// Number of payload bytes observed in the flow. Does not include bytes from malformed
265    /// segments.
266    pub nb_payload_bytes: u64,
267    /// Maximum number of simultaneous content gaps.
268    ///
269    /// A content gap is a "hole" in the TCP sequence number, indicated re-ordered or missing
270    /// packets. Only applies to TCP flows.
271    pub max_simult_gaps: u64,
272    /// Starting sequence number of the first byte in the first payload (ISN + 1). Only applies to
273    /// TCP flows, and is set to `0` for UDP.
274    pub data_start: u32,
275    /// Maximum chunk capacity (the maximum number of simultaneous gaps + 1). Only applies to TCP
276    /// flows. Used to prevent resizing `chunks` vector.
277    pub capacity: usize,
278    /// The set of non-overlapping content intervals. Only applies to TCP flows.
279    pub chunks: Vec<Chunk>,
280    /// Maps relative sequence number of a content gap to the number of packets observed before it
281    /// is filled. Only applies to TCP flows.
282    pub gaps: HashMap<u32, u64>,
283}
284
285impl Flow {
286    fn new() -> Self {
287        Flow {
288            nb_pkts: 0,
289            nb_malformed_pkts: 0,
290            nb_late_start_pkts: 0,
291            nb_pkt_bytes: 0,
292            nb_payload_bytes: 0,
293            max_simult_gaps: 0,
294            data_start: 0,
295            capacity: DEFAULT_CHUNK_CAPACITY,
296            chunks: Vec::with_capacity(DEFAULT_CHUNK_CAPACITY),
297            gaps: HashMap::new(),
298        }
299    }
300
301    #[inline]
302    fn insert_segment(&mut self, segment: &L4Pdu) {
303        self.nb_pkts += 1;
304        self.nb_pkt_bytes += segment.mbuf.data_len() as u64;
305
306        if segment.offset() > segment.mbuf.data_len()
307            || (segment.offset() + segment.length()) > segment.mbuf.data_len()
308        {
309            self.nb_malformed_pkts += 1;
310            return;
311        }
312        self.nb_payload_bytes += segment.length() as u64;
313
314        let seq_no = if segment.flags() & SYN != 0 {
315            segment.seq_no().wrapping_add(1)
316        } else {
317            segment.seq_no()
318        };
319
320        if self.chunks.is_empty() {
321            self.data_start = seq_no;
322        }
323
324        if wrapping_lt(seq_no, self.data_start) {
325            self.nb_late_start_pkts += 1;
326            return;
327        }
328
329        if self.chunks.len() < self.capacity {
330            let seg_start = seq_no.wrapping_sub(self.data_start);
331            let seg_end = seg_start + segment.length() as u32;
332
333            self.merge_chunk(Chunk(seg_start, seg_end));
334        }
335    }
336
337    /// Insert `chunk` into flow, merging intervals as necessary. Flow `chunks` are a sorted set of
338    /// non-overlapping intervals.
339    #[inline]
340    fn merge_chunk(&mut self, chunk: Chunk) {
341        let mut start = chunk.0;
342        let mut end = chunk.1;
343
344        let mut result = vec![];
345        let mut inserted = false;
346        for chunk in self.chunks.iter() {
347            if inserted || start > chunk.1 {
348                result.push(*chunk);
349            } else if end < chunk.0 {
350                inserted = true;
351                result.push(Chunk(start, end));
352                result.push(*chunk);
353            } else {
354                start = std::cmp::min(start, chunk.0);
355                end = std::cmp::max(end, chunk.1);
356            }
357        }
358        if !inserted {
359            result.push(Chunk(start, end));
360        }
361
362        for chunk in result[..result.len() - 1].iter() {
363            *self.gaps.entry(chunk.1).or_insert(0) += 1;
364        }
365
366        if result.len().saturating_sub(1) as u64 > self.max_simult_gaps {
367            self.max_simult_gaps += 1;
368        }
369        self.chunks = result;
370    }
371
372    /// Returns the number of content gaps at the connection end.
373    ///
374    /// This is not the total number of content gaps ever observed, rather, it represents the total
375    /// number of gaps remaining in the final state of the connection.
376    #[inline]
377    pub fn content_gaps(&self) -> u64 {
378        self.chunks.len().saturating_sub(1) as u64
379    }
380
381    /// Number of bytes missed in content gaps at connection end.
382    ///
383    /// This is not the total size of all content gaps ever observed, rather, it represents the
384    /// total number of missing bytes in the final state of the connection.
385    #[inline]
386    pub fn missed_bytes(&self) -> u64 {
387        self.chunks.windows(2).map(|w| w[1].0 - w[0].1).sum::<u32>() as u64
388    }
389
390    /// Returns the mean number of packet arrivals before a content gap is filled, or `0` if there
391    /// were no gaps.
392    #[inline]
393    pub fn mean_pkts_to_fill(&self) -> Option<f64> {
394        if self.gaps.is_empty() {
395            return None;
396        }
397        let mut sum = 0;
398        for val in self.gaps.values() {
399            sum += *val;
400        }
401        Some(sum as f64 / self.gaps.len() as f64)
402    }
403
404    /// Returns the median number of packet arrivals before a content gap is filled, or `0` if there
405    /// were no gaps.
406    #[inline]
407    pub fn median_pkts_to_fill(&self) -> Option<u64> {
408        if self.gaps.is_empty() {
409            return None;
410        }
411        let mut values = self.gaps.values().collect::<Vec<_>>();
412        values.sort();
413        let mid = values.len() / 2;
414        Some(*values[mid])
415    }
416}
