1//! Vectors of raw packet data.
2//! All of these types are Connection-level, meaning they are delivered on
3//! connection termination.
4//!
5//! There are two types of packet lists: those containing Mbufs (`Zc` prefix)
6//! and those containing raw bytes (as vectors).
7//!
8//! For the former: Mbufs are shared across these lists via DPDK reference counting,
9//! so requesting lists of packets does not require copying. However, it may
10//! introduce additional mempool requirements, as Mbufs must be kept in memory
11//! for the duration of the connection. This is often particularly infeasible for
12//! UDP connections, which must stay in memory until a timeout is reached.
13//! In such cases, users may wish to use the non-`Zc` variants.
14//!
15//! For TCP connections, the non-`Zc` variants wait to clone data until the
16//! first few packets have passed or until the packet data is requested.
17//! After the first `PKTS_START_CLONE` packets, it is likely that some traffic has
18//! been filtered out by the framework (e.g., TLS handshake has been parsed).
19//! This is a middle ground between memory usage and compute performance.
20//!
21//! For UDP connections, this is not feasible without unacceptable mempool utilization;
22//! many UDP connections are short-lived, and UDP connections are not "closed" until
23//! a timeout period has passed.
24

30/// Pasic raw packet bytes.
31#[derive(Debug)]
32pub struct PktData {
33    pub data: Vec<u8>,
34}
35
36impl PktData {
37    pub fn new(mbuf: &Mbuf) -> Self {
38        Self {
39            data: mbuf.data().to_vec(),
40        }
41    }
42}
43
44/// Number of Mbufs to cache before starting to clone data for
45/// TCP connections only. If PKTS_START_CLONE is not reached, the
46/// data is converted to Vec<u8> on first access.
47const PKTS_START_CLONE: usize = 5;
48
49pub trait PktStream {
50    fn in_mbufs_own(&mut self) -> Vec<Mbuf>;
51    fn in_mbufs_ref(&mut self) -> &mut Vec<Mbuf>;
52    fn out_packets(&mut self) -> &mut Vec<PktData>;
53
54    fn drain_mbufs(&mut self) {
55        let mut in_mbufs = self.in_mbufs_own();
56        for mbuf in in_mbufs.drain(..) {
57            self.out_packets().push(PktData::new(&mbuf));
58        }
59    }
60
61    fn packets(&mut self) -> &Vec<PktData> {
62        if self.in_mbufs_ref().is_empty() {
63            return self.out_packets();
64        }
65        self.drain_mbufs();
66        self.out_packets()
67    }
68
69    fn push(&mut self, pdu: &L4Pdu) {
70        if pdu.ctxt.proto == TCP_PROTOCOL && self.in_mbufs_ref().len() < PKTS_START_CLONE {
71            self.in_mbufs_ref().push(Mbuf::new_ref(&pdu.mbuf));
72            return;
73        } else if !self.in_mbufs_ref().is_empty() {
74            self.drain_mbufs();
75        }
76        self.out_packets().push(PktData::new(pdu.mbuf_ref()));
77    }
78}

80/// For a connection, the bidirectional stream of packets
81/// in the order received by the framework.
82#[derive(Debug)]
83#[cfg_attr(not(feature = "skip_expand"), datatype)]
84pub struct BidirPktStream {
85    /// The raw packet data.
86    pub packets: Vec<PktData>,
87    /// The first few packets are stored as Mbufs
88    /// before data copies begin.
89    mbufs: Vec<Mbuf>,
90}
91
92impl PktStream for BidirPktStream {
93    fn in_mbufs_own(&mut self) -> Vec<Mbuf> {
94        std::mem::take(&mut self.mbufs)
95    }
96
97    fn in_mbufs_ref(&mut self) -> &mut Vec<Mbuf> {
98        &mut self.mbufs
99    }
100
101    fn out_packets(&mut self) -> &mut Vec<PktData> {
102        &mut self.packets
103    }
104}
105
106impl BidirPktStream {
107    #[cfg_attr(
108        not(feature = "skip_expand"),
109        datatype_fn("BidirPktStream,level=InL4Conn")
110    )]
111    pub fn update(&mut self, pdu: &L4Pdu) {
112        self.push(pdu);
113    }
114}
115
116impl Tracked for BidirPktStream {
117    fn new(_first_pkt: &L4Pdu) -> Self {
118        Self {
119            packets: Vec::new(),
120            mbufs: Vec::new(),
121        }
122    }
123
124    fn clear(&mut self) {
125        self.packets.clear();
126        self.mbufs.clear();
127    }
128}
