24use std::collections::hash_map::DefaultHasher;
25use std::hash::{Hash, Hasher};
26use std::net::{IpAddr, Ipv4Addr, Ipv6Addr};
27use std::time::Instant;
28
29// /24 and /64
30const V4_MASK: u32 = u32::MAX << (32 - 24);
31const V6_BYTES: usize = 8;
32const V6_BYTES_LEN: usize = 16;
33
34#[derive(Debug)]
35pub struct AnonFiveTuple {
36    pub data: FiveTuple,
37}
38
39impl AnonFiveTuple {
40    fn hash_ip(ip: &IpAddr) -> IpAddr {
41        match ip {
42            IpAddr::V4(addr) => {
43                let octets = addr.octets();
44                let addr_u32 = u32::from_be_bytes(octets);
45                let mut hasher = DefaultHasher::new();
46                addr.hash(&mut hasher);
47                let hash = hasher.finish() as u32;
48                IpAddr::V4(Ipv4Addr::from((addr_u32 & V4_MASK) | (hash & !V4_MASK)))
49            }
50            IpAddr::V6(addr) => {
51                let mut hasher = DefaultHasher::new();
52                addr.hash(&mut hasher);
53                let hash = hasher.finish().to_be_bytes();
54                let mut addr = addr.octets();
55                for i in V6_BYTES..V6_BYTES_LEN {
56                    addr[i] = hash[i % hash.len()];
57                }
58                IpAddr::V6(Ipv6Addr::from(addr))
59            }
60        }
61    }
62}
63
64impl StaticData for AnonFiveTuple {
65    #[cfg_attr(
66        not(feature = "skip_expand"),
67        datatype("name=AnonFiveTuple,level=L4FirstPacket")
68    )]
69    fn new(first_pkt: &L4Pdu) -> Self {
70        let mut five_tuple = FiveTuple::from_ctxt(&first_pkt.ctxt);
71        five_tuple.orig.set_ip(Self::hash_ip(&five_tuple.orig.ip()));
72        Self { data: five_tuple }
73    }
74}

76/// Five-tuple with LSBs cleared for anonymization.
77/// This preserves subnet structure.
78#[derive(Debug)]
79pub struct ClearedFiveTuple {
80    pub data: FiveTuple,
81}
82
83impl ClearedFiveTuple {
84    fn clear(ip: IpAddr) -> IpAddr {
85        match ip {
86            IpAddr::V4(addr) => {
87                let addr = u32::from_be_bytes(addr.octets());
88                IpAddr::V4(Ipv4Addr::from(addr & V4_MASK))
89            }
90            IpAddr::V6(addr) => {
91                let mut addr = addr.octets();
92                for i in 1..V6_BYTES + 1 {
93                    addr[V6_BYTES_LEN - i] = 0;
94                }
95                IpAddr::V6(Ipv6Addr::from(addr))
96            }
97        }
98    }
99}

101impl StaticData for ClearedFiveTuple {
102    #[cfg_attr(
103        not(feature = "skip_expand"),
104        datatype("name=ClearedFiveTuple,level=L4FirstPacket")
105    )]
106    fn new(first_pkt: &L4Pdu) -> Self {
107        let mut five_tuple = FiveTuple::from_ctxt(&first_pkt.ctxt);
108        five_tuple.orig.set_ip(Self::clear(five_tuple.orig.ip()));
109        five_tuple.resp.set_ip(Self::clear(five_tuple.resp.ip()));
110        Self { data: five_tuple }
111    }
112}

128/// Tag Control Information field on the first packet, or none
129#[derive(Clone, Copy, Debug, PartialEq)]
130pub struct EtherTCI(Option<u16>);
131
132impl StaticData for EtherTCI {
133    #[cfg_attr(
134        not(feature = "skip_expand"),
135        datatype("name=EtherTCI,level=L4FirstPacket")
136    )]
137    fn new(first_pkt: &L4Pdu) -> Self {
138        if let Ok(ethernet) = &Packet::parse_to::<Ethernet>(first_pkt.mbuf_ref()) {
139            if let Some(tci) = ethernet.tci() {
140                return EtherTCI(Some(tci));
141            }
142        }
143        EtherTCI(None)
144    }
145}

147/// The src/dst MAC of a connection
148#[derive(Clone, Debug)]
149pub struct EthAddr {
150    pub src: MacAddr,
151    pub dst: MacAddr,
152}
153
154impl StaticData for EthAddr {
155    #[cfg_attr(
156        not(feature = "skip_expand"),
157        datatype("name=EthAddr,level=L4FirstPacket")
158    )]
159    fn new(first_pkt: &L4Pdu) -> Self {
160        if let Ok(ethernet) = &Packet::parse_to::<Ethernet>(first_pkt.mbuf_ref()) {
161            Self {
162                src: ethernet.src(),
163                dst: ethernet.dst(),
164            }
165        } else {
166            panic!("Non-ethernet packets not supported");
167        }
168    }
